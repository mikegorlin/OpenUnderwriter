    2. Create `src/do_uw/stages/render/facet_renderer.py` (~100 lines):
       ```python
       """Facet-driven rendering orchestrator.

       Builds facet dispatch context for HTML templates. Facets with
       subsections are rendered via facet dispatch; facets without
       subsections fall through to legacy hardcoded templates.

       Phase 56 scope: HTML rendering only. Word/Markdown untouched.
       """
       from __future__ import annotations

       import logging
       from pathlib import Path
       from typing import Any

       from do_uw.brain.brain_facet_schema import FacetSpec, load_all_facets
       from do_uw.models.state import AnalysisState

       logger = logging.getLogger(__name__)

       _BRAIN_FACETS_DIR = Path(__file__).parent.parent.parent / "brain" / "facets"


       def build_facet_context(
           state: AnalysisState | None = None,
           facets_dir: Path | None = None,
       ) -> dict[str, Any]:
           """Build facet dispatch context for HTML rendering.

           Returns dict with facet_sections: dict of facet_id -> subsection data.
           Only facets with non-empty subsections are included (legacy facets excluded).

           Args:
               state: AnalysisState (reserved for future per-signal gating).
               facets_dir: Override facets directory (for testing).
           """
           _ = state  # Reserved for future use
           fdir = facets_dir or _BRAIN_FACETS_DIR
           facets = load_all_facets(fdir)
           facet_sections: dict[str, Any] = {}

           for facet_id, facet in facets.items():
               if not facet.subsections:
                   continue  # Legacy rendering -- skip

               subsection_data = []
               for sub in facet.subsections:
                   template = sub.template or f"sections/financial/{sub.id}.html.j2"
                   subsection_data.append({
                       "id": sub.id,
                       "name": sub.name,
                       "render_as": sub.render_as,
                       "signals": sub.signals,
                       "columns": sub.columns,
                       "template": template,
                   })

               facet_sections[facet_id] = {
                   "facet": facet,
                   "subsections": subsection_data,
               }
               logger.debug(
                   "Facet '%s' has %d subsections for facet-driven rendering",
                   facet_id,
                   len(subsection_data),
               )

           return {"facet_sections": facet_sections}
       ```
