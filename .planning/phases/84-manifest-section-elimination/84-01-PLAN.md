        """Map group IDs to lists of signal IDs from loaded signals."""
        groups: dict[str, list[str]] = {}
        for sig in signals:
            gid = sig.get("group", "")
            if gid:
                groups.setdefault(gid, []).append(sig["id"])
        return groups
