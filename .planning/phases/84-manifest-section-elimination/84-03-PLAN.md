       ```python
       def _build_signal_section_map() -> tuple[dict[str, str], dict[str, str]]:
           """Build signal-to-prefix and prefix-to-name maps from manifest + signal groups."""
           from do_uw.brain.manifest_schema import load_manifest
           from do_uw.brain.brain_unified_loader import load_signals
           manifest = load_manifest()
           signals_data = load_signals()

           # Build group_id -> (section_prefix, section_name)
           group_to_section: dict[str, tuple[str, str]] = {}
           for ms in manifest.sections:
               # Section prefix = first part of section.id or derive from manifest
               # Use section.id as the section prefix for grouping
               for group in ms.groups:
                   group_to_section[group.id] = (ms.id, ms.name)

           signal_to_prefix: dict[str, str] = {}
           prefix_to_name: dict[str, str] = {}
           for sig in signals_data.get("signals", []):
               gid = sig.get("group", "")
               if gid and gid in group_to_section:
                   sec_id, sec_name = group_to_section[gid]
                   signal_to_prefix[sig["id"]] = sec_id
                   prefix_to_name[sec_id] = sec_name
               elif "." in sig.get("id", ""):
                   # Fallback: infer from ID prefix
                   prefix = sig["id"].split(".")[0]
                   signal_to_prefix[sig["id"]] = prefix
           return signal_to_prefix, prefix_to_name
       ```

       ```python
           """Look up group_id and group_name for a signal from manifest."""
           from do_uw.brain.manifest_schema import load_manifest
           from do_uw.brain.brain_unified_loader import load_signals

           for sig in signals_data.get("signals", []):

                   if gid:
       ```
