    **Update extract_do_implications** to call generators instead of returning static text:
    ```python
    def extract_do_implications(state):
        result = {}
        for section_id, conditions in _DO_IMPLICATIONS_REGISTRY.items():
            items = []
            for cond_key, severity in conditions:
                if check_implication_condition(state, section_id, cond_key):
                    gen = _IMPLICATION_GENERATORS.get(cond_key)
                    if gen:
                        text = gen(state)
                        if text:
                            items.append({"text": text, "severity": severity})
            if items:
                tpl_key = _TEMPLATE_KEYS.get(section_id, section_id)
                result[tpl_key] = {
                    "items": items,
                    "coverage_note": _gen_coverage_note(state, section_id),
                }
        return result
    ```
