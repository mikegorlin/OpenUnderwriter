def ohlson_do_context(score: float | None, zone: str | None) -> str:
    if score is None or zone is None:
        return ""
    if zone == "distress" or (score is not None and score > 0.5):
        return (
            "Elevated bankruptcy probability (O-Score > 0.5). "
            "The Ohlson model considers size, leverage, profitability, and "
            "liquidity — high scores correlate with Zone-of-Insolvency "
            "fiduciary duty claims against directors."
        )
    return (
        "Low bankruptcy probability — supportive of favorable D&O risk profile."
    )
