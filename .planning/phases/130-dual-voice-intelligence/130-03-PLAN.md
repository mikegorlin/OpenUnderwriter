```jinja2
{# Dual-voice commentary block macro (Phase 130).
   Renders "What Was Said" (factual) and "Underwriting Commentary" (D&O analysis)
   as side-by-side visually distinct blocks.

   Parameters:
     comm: dict with keys: what_was_said, underwriting_commentary, confidence, hallucination_warnings
     section_name: human-readable section name for accessibility
#}
{% macro dual_voice_block(comm, section_name="") %}
{% if comm and (comm.what_was_said or comm.underwriting_commentary) %}
<div class="dual-voice" role="region" aria-label="{{ section_name }} Commentary">
  <div class="dual-voice__factual">
    <div class="dual-voice__header">What Was Said</div>
    <div class="dual-voice__body">{{ comm.what_was_said | safe }}</div>
  </div>
  <div class="dual-voice__commentary">
    <div class="dual-voice__header">
      Underwriting Commentary
      {% if comm.confidence %}
      <span class="confidence-badge confidence-badge--{{ comm.confidence | lower }}">
        {{ comm.confidence }}
      </span>
      {% endif %}
    </div>
    <div class="dual-voice__body">{{ comm.underwriting_commentary | safe }}</div>
  </div>
</div>
{% endif %}
{% endmacro %}
```

```python
"""Commentary context builder for dual-voice template rendering (Phase 130).

Reads PreComputedCommentary from state and makes it available to
section templates as `commentary` dict.
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import Any

from do_uw.models.state import AnalysisState
from do_uw.stages.render.context_builders.assembly_registry import (
    register_builder,
)

logger = logging.getLogger(__name__)


def _serialize_commentary(comm: Any) -> dict[str, Any] | None:
    """Serialize a SectionCommentary model to a template-friendly dict."""
    if comm is None:
        return None
    return {
        "what_was_said": comm.what_was_said or "",
        "underwriting_commentary": comm.underwriting_commentary or "",
        "confidence": comm.confidence or "MEDIUM",
        "hallucination_warnings": list(comm.hallucination_warnings or []),
    }


@register_builder
def _build_commentary_context(
    state: AnalysisState,
    context: dict[str, Any],
    chart_dir: Path | None,
) -> None:
    """Populate commentary context for dual-voice template blocks."""
    if (
        state.analysis is None
        or state.analysis.pre_computed_commentary is None
    ):
        context["commentary"] = {}
        return

    pcc = state.analysis.pre_computed_commentary
    context["commentary"] = {
        "executive_brief": _serialize_commentary(pcc.executive_brief),
        "financial": _serialize_commentary(pcc.financial),
        "market": _serialize_commentary(pcc.market),
        "governance": _serialize_commentary(pcc.governance),
        "litigation": _serialize_commentary(pcc.litigation),
        "scoring": _serialize_commentary(pcc.scoring),
        "company": _serialize_commentary(pcc.company),
        "meeting_prep": _serialize_commentary(pcc.meeting_prep),
    }
    logger.info(
        "Commentary context: %d sections populated",
        sum(1 for v in context["commentary"].values() if v is not None),
    )
```
