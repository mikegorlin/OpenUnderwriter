```python
def build_hae_context(state: AnalysisState) -> dict[str, Any]:
    """Extract H/A/E composite scores for radar chart rendering."""
    if state.scoring is None:
        return {"hae_available": False}
    hae_result = getattr(state.scoring, "hae_result", None)
    if hae_result is None:
        return {"hae_available": False}

    return {
        "hae_available": True,
        "host_composite": hae_result.host_composite,
        "agent_composite": hae_result.agent_composite,
        "environment_composite": hae_result.environment_composite,
        "probability": hae_result.probability,
        "tier": hae_result.tier,
        "crf_vetoes": hae_result.crf_vetoes if hasattr(hae_result, 'crf_vetoes') else [],
        # Radar chart data points (3 axes)
        "radar_labels": ["Host (Structural)", "Agent (Behavioral)", "Environment (External)"],
        "radar_values": [
            hae_result.host_composite,
            hae_result.agent_composite,
            hae_result.environment_composite,
        ],
    }
```
