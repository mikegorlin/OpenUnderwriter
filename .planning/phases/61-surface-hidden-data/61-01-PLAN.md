   ```python
   def extract_peer_matrix(state: AnalysisState) -> dict[str, Any] | None:
       """Extract full peer comparison matrix for template.

       Builds a table with company + peers as rows and benchmark metrics as columns.
       Each cell includes value, percentile rank, and source/confidence.
       """
       bm = state.benchmark
       if bm is None or not bm.metric_details:
           return None

       # Build metric columns from metric_details
       metrics = []
       for name, mb in bm.metric_details.items():
           metrics.append({
               "name": mb.metric_name,
               "company_value": format_value(mb.company_value, mb.metric_name),
               "percentile_rank": mb.percentile_rank,
               "peer_count": mb.peer_count,
               "baseline_value": format_value(mb.baseline_value, mb.metric_name),
               "higher_is_better": mb.higher_is_better,
               "section": mb.section,
           })

       # Build peer rows from peer_quality_scores
       peers = []
       for ticker, qscore in bm.peer_quality_scores.items():
           peers.append({
               "ticker": ticker,
               "quality_score": f"{qscore:.0f}",
           })

       return {
           "metrics": metrics,
           "peers": peers,
           "peer_tickers": bm.peer_group_tickers,
           "relative_position": bm.relative_position or "N/A",
           "sector_average": f"{bm.sector_average_score:.0f}" if bm.sector_average_score else "N/A",
           "peer_rankings": {k: f"{v:.0f}" for k, v in bm.peer_rankings.items()},
       }
   ```
