    ```jinja2
    {# Paired-column KV table: 4 cols/row (Label|Value|Label|Value) for CIQ density.
       pairs: list of (key, value) tuples or dicts with 'key' and 'value'.
       Renders pairs in 2x2 grid: items 0,1 on row 1; items 2,3 on row 2; etc.
       title: optional table title shown as a header row. #}
    {% macro paired_kv_table(pairs, title="") %}
    <table class="w-full border-collapse my-3 text-sm kv-table kv-paired">
      {% if title %}
      <thead>
        <tr>
          <th colspan="4" class="bg-slate-100 text-navy font-heading text-xs font-semibold uppercase tracking-wider text-left px-3 py-2 border-b border-gray-200">
            {{ title }}
          </th>
        </tr>
      </thead>
      {% endif %}
      <tbody>
        {% for i in range(0, pairs|length, 2) %}
        {%- set p1 = pairs[i] -%}
        {%- set k1 = p1.key if p1 is mapping else p1[0] -%}
        {%- set v1 = p1.value if p1 is mapping else p1[1] -%}
        <tr class="{{ 'bg-bg-alt' if loop.index is odd else 'bg-white' }}">
          <th class="w-[18%] text-left px-3 py-2 font-heading text-xs font-semibold text-navy border-b border-gray-200 align-top">{{ k1 }}</th>
          <td class="w-[32%] px-3 py-2 border-b border-gray-200 align-top">{{ v1 | format_na }}</td>
          {% if i + 1 < pairs|length %}
            {%- set p2 = pairs[i + 1] -%}
            {%- set k2 = p2.key if p2 is mapping else p2[0] -%}
            {%- set v2 = p2.value if p2 is mapping else p2[1] -%}
            <th class="w-[18%] text-left px-3 py-2 font-heading text-xs font-semibold text-navy border-b border-gray-200 border-l border-gray-100 align-top">{{ k2 }}</th>
            <td class="w-[32%] px-3 py-2 border-b border-gray-200 align-top">{{ v2 | format_na }}</td>
          {% else %}
            <th class="w-[18%] border-b border-gray-200"></th>
            <td class="w-[32%] border-b border-gray-200"></td>
          {% endif %}
        </tr>
        {% endfor %}
      </tbody>
    </table>
    {% endmacro %}
    ```
