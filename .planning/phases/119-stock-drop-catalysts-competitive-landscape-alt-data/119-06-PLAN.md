**CRITICAL: Use explicit AnalysisState fields, NOT underscore-prefixed attributes.**

Pydantic v2 with `ConfigDict(frozen=False)` and no `extra` config defaults to `extra='ignore'`. This means `state._foo = bar` is SILENTLY DROPPED -- the value is never stored. Plan 01 adds explicit fields: `state.stock_patterns`, `state.multi_horizon_returns`, `state.analyst_consensus`, `state.drop_narrative`. Use THESE, not `state._stock_patterns` etc.

            # Phase 15: Stock drop catalyst enrichment (Phase 119)
            try:
                from do_uw.stages.extract.stock_catalyst import (
                    enrich_drops_with_prices_and_volume,
                    detect_stock_patterns,
                )
                from do_uw.stages.extract.stock_performance_summary import (
                    compute_multi_horizon_returns,
                    build_analyst_consensus,
                )
                mkt = state.extracted.market
                if mkt and mkt.stock_drops:
                    all_drops = (mkt.stock_drops.single_day_drops or []) + (mkt.stock_drops.multi_day_drops or [])
                    history = getattr(state.acquired_data, "market_data", {}) or {}
                    enrich_drops_with_prices_and_volume(all_drops, history.get("history_2y", {}))
                    state.stock_patterns = detect_stock_patterns(
                        all_drops,
                        trading_days_available=getattr(mkt.stock_performance, "trading_days_available", None) if mkt.stock_performance else None,
                    )
                # Multi-horizon returns
                if mkt and mkt.stock_performance:
                    from do_uw.stages.extract.stock_drops import get_close_prices
                    history_data = (getattr(state.acquired_data, "market_data", {}) or {}).get("history_1y", {})
                    prices = get_close_prices(history_data)
                    td = getattr(mkt.stock_performance, "trading_days_available", None)
                    state.multi_horizon_returns = compute_multi_horizon_returns(prices, trading_days_available=td)
                # Analyst consensus
                if mkt and mkt.analyst_sentiment:
                    market_data = getattr(state.acquired_data, "market_data", {}) or {}
                    current = mkt.stock_performance.current_price.value if mkt.stock_performance and mkt.stock_performance.current_price else None
                    state.analyst_consensus = build_analyst_consensus(mkt.analyst_sentiment, market_data, current)
                logger.info("Phase 15: Stock catalyst enrichment complete")
            except Exception:
                logger.warning("Phase 15: Stock catalyst enrichment failed", exc_info=True)

            # Phase 16: Alt data extraction (Phase 119)
            try:
                from do_uw.stages.extract.alt_data_extraction import extract_alt_data
                extract_alt_data(state)
                logger.info("Phase 16: Alt data extraction complete")
            except Exception:
                logger.warning("Phase 16: Alt data extraction failed", exc_info=True)

            # Phase 17: Competitive landscape extraction (Phase 119)
            try:
                from do_uw.stages.extract.competitive_extraction import extract_competitive_landscape
                await extract_competitive_landscape(state)
                logger.info("Phase 17: Competitive landscape extraction complete")
            except Exception:
                logger.warning("Phase 17: Competitive landscape extraction failed", exc_info=True)
