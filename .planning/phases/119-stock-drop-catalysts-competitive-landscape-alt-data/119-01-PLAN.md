class PeerRow(BaseModel):
    model_config = ConfigDict(frozen=False)
    company_name: str = Field(default="")
    ticker: str = Field(default="")
    market_cap: str = Field(default="")
    revenue: str = Field(default="")
    margin: str = Field(default="")
    growth_rate: str = Field(default="")
    rd_spend: str = Field(default="")
    market_share: str = Field(default="")
    stock_performance: str = Field(default="")
    sca_history: str = Field(default="")  # SCA filings against this peer
    do_relevance: str = Field(default="")

class MoatDimension(BaseModel):
    model_config = ConfigDict(frozen=False)
    dimension: str = Field(default="")  # "Data Advantage", "Switching Costs", etc.
    present: bool = Field(default=False)
    strength: str = Field(default="")  # "Strong", "Moderate", "Weak"
    durability: str = Field(default="")  # "High", "Medium", "Low"
    evidence: str = Field(default="")
    do_risk: str = Field(default="")  # D&O risk if moat erodes

class CompetitiveLandscape(BaseModel):
    model_config = ConfigDict(frozen=False)
    peers: list[PeerRow] = Field(default_factory=list)
    moat_dimensions: list[MoatDimension] = Field(default_factory=list)
    competitive_position_narrative: str = Field(default="")
    do_commentary: str = Field(default="")  # Overall competitive D&O risk narrative
```

class ESGRisk(BaseModel):
    model_config = ConfigDict(frozen=False)
    controversies: list[str] = Field(default_factory=list)
    ratings: dict[str, str] = Field(default_factory=dict)  # {"MSCI": "BBB", "ISS": "3"}
    greenwashing_indicators: list[str] = Field(default_factory=list)
    mission_alignment: str = Field(default="")
    risk_level: str = Field(default="LOW")
    do_relevance: str = Field(default="")

class AIWashingRisk(BaseModel):
    model_config = ConfigDict(frozen=False)
    ai_claims_present: bool = Field(default=False)
    indicators: list[dict[str, str]] = Field(default_factory=list)  # [{"claim": "...", "evidence": "...", "risk": "..."}]
    scienter_risk: str = Field(default="LOW")
    do_relevance: str = Field(default="")

class TariffExposure(BaseModel):
    model_config = ConfigDict(frozen=False)
    supply_chain_exposure: str = Field(default="")
    manufacturing_locations: list[str] = Field(default_factory=list)
    international_revenue_pct: str = Field(default="")
    tariff_risk_factors: list[str] = Field(default_factory=list)
    risk_level: str = Field(default="LOW")
    do_relevance: str = Field(default="")

class PeerSCACheck(BaseModel):
    model_config = ConfigDict(frozen=False)
    peer_scas: list[dict[str, str]] = Field(default_factory=list)  # [{"company": "...", "filing_date": "...", "allegation": "..."}]
    sector: str = Field(default="")
    contagion_risk: str = Field(default="LOW")
    do_relevance: str = Field(default="")

class AltDataAssessments(BaseModel):
    model_config = ConfigDict(frozen=False)
    esg: ESGRisk = Field(default_factory=ESGRisk)
    ai_washing: AIWashingRisk = Field(default_factory=AIWashingRisk)
    tariff: TariffExposure = Field(default_factory=TariffExposure)
    peer_sca: PeerSCACheck = Field(default_factory=PeerSCACheck)
```
