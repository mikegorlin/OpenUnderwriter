```python
"""Phase 119: Competitive landscape D&O enrichment."""
from __future__ import annotations

from do_uw.models.state import AnalysisState


def enrich_competitive_landscape(state: AnalysisState) -> None:
    """Add D&O commentary to competitive landscape data."""
    cl = state.dossier.competitive_landscape
    if not cl.peers and not cl.moat_dimensions:
        return

    company = state.company.name or state.company.ticker or "Company"

    # Build do_commentary from peers + moat dimensions
    peer_count = len(cl.peers)
    moat_present = [m for m in cl.moat_dimensions if m.present]
    moat_weak = [m for m in cl.moat_dimensions if m.present and m.strength == "Weak"]

    parts = [f"{company} faces competition from {peer_count} identified peers."]
    if moat_present:
        parts.append(f"{len(moat_present)} moat dimensions present ({', '.join(m.dimension for m in moat_present)}).")
    if moat_weak:
        parts.append(f"WARNING: {len(moat_weak)} moat(s) rated Weak — erosion risk elevated.")
    cl.do_commentary = " ".join(parts)

    # Per-moat do_risk
    _MOAT_DO_RISK = {
        "Scale Economics": "Scale erosion triggers revenue compression → guidance miss → 10(b) claims",
        "Switching Costs": "Low switching costs = customer churn risk → revenue volatility → SCA exposure",
        "Brand Premium": "Brand damage (product recall, scandal) → stock drop → D&O claims",
        "Network Effects": "Network shrinkage accelerates → user/revenue loss → Section 10(b) exposure",
        "Data Advantage": "Data breach or regulatory restriction → competitive loss → stock decline",
        "Regulatory Barrier": "Deregulation removes barrier → margin compression → guidance failure",
        "Distribution Lock": "Channel disruption → revenue concentration loss → earnings miss",
    }
    for moat in cl.moat_dimensions:
        if moat.present and moat.strength in ("Weak", "Moderate"):
            moat.do_risk = _MOAT_DO_RISK.get(moat.dimension, f"{moat.dimension} erosion creates D&O exposure")

"""Phase 119: Alternative data D&O relevance enrichment."""
from __future__ import annotations

from do_uw.models.state import AnalysisState


def enrich_alt_data(state: AnalysisState) -> None:
    """Add D&O relevance narratives to all alt data assessments."""
    company = state.company.name or state.company.ticker or "Company"
    ad = state.alt_data

    # ESG
    if ad.esg.risk_level != "LOW" or ad.esg.controversies:
        controv_count = len(ad.esg.controversies)
        ad.esg.do_relevance = (
            f"{company} ESG risk: {ad.esg.risk_level}. "
            f"{controv_count} controversies identified. "
            "ESG gaps create shareholder derivative suit exposure under Caremark "
            "duty of oversight. Greenwashing claims trigger SEC enforcement + "
            "consumer class actions that compound D&O exposure."
        )
    else:
        ad.esg.do_relevance = f"{company} ESG profile: LOW risk. No significant controversies identified."

    # AI-Washing
    if ad.ai_washing.ai_claims_present:
        indicator_count = len(ad.ai_washing.indicators)
        ad.ai_washing.do_relevance = (
            f"{company} makes AI-related claims ({indicator_count} indicators). "
            f"Scienter risk: {ad.ai_washing.scienter_risk}. "
            "AI claims without substantive evidence create 10(b) fraud exposure — "
            "SEC has signaled AI-washing enforcement priority. "
            "Plaintiff attorneys pattern-match AI claims → stock drop → SCA filing."
        )
    else:
        ad.ai_washing.do_relevance = f"No AI-related claims detected for {company}."

    # Tariff
    if ad.tariff.risk_level != "LOW":
        ad.tariff.do_relevance = (
            f"{company} tariff exposure: {ad.tariff.risk_level}. "
            f"Supply chain: {ad.tariff.supply_chain_exposure or 'Unknown'}. "
            f"International revenue: {ad.tariff.international_revenue_pct or 'N/A'}. "
            "Tariff escalation threatens revenue guidance accuracy — "
            "management failure to disclose known supply chain risk "
            "creates Section 10(b) omission liability."
        )
    else:
        ad.tariff.do_relevance = f"{company} tariff exposure: LOW. Limited international/supply chain risk."

    # Peer SCA
    peer_count = len(ad.peer_sca.peer_scas)
    if peer_count > 0:
        ad.peer_sca.do_relevance = (
            f"{peer_count} active SCA(s) against {company}'s sector peers. "
            f"Contagion risk: {ad.peer_sca.contagion_risk}. "
            "Sector SCA activity indicates plaintiff attorney focus on this industry — "
            "successful peer settlements establish precedent and attract copycat filings."
        )
    else:
        ad.peer_sca.do_relevance = f"No active SCAs against {company}'s sector peers. Low contagion risk."
```
