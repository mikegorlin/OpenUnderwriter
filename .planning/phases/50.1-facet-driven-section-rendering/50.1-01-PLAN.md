   ```python
   def _build_signal_section_map() -> tuple[dict[str, str], dict[str, str]]:
       """Build signal-to-prefix and prefix-to-name maps from facet definitions.

       Returns (signal_to_prefix, prefix_to_name) where:
       - signal_to_prefix maps each signal ID to its facet's prefix (e.g. "BIZ")
       - prefix_to_name maps each prefix to its facet display name (e.g. "Business Profile")
       """
       facets = _get_facets()
       signal_to_prefix: dict[str, str] = {}
       prefix_to_name: dict[str, str] = {}
       for facet in facets.values():
           if not facet.signals:
               continue  # Skip red_flags (empty signals list)
           facet_prefix = facet.signals[0].split(".")[0]
           prefix_to_name[facet_prefix] = facet.name
           for sig_id in facet.signals:
               signal_to_prefix[sig_id] = facet_prefix
       return signal_to_prefix, prefix_to_name
   ```
