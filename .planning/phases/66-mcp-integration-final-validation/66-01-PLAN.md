       ```python
       # Phase B++++: CourtListener federal case search (non-blocking).
       try:
           cl_results = self._courtlistener_client.search_cases(
               company_name, state.ticker, cache=self._cache
           )
           if cl_results:
               acquired.litigation_data["courtlistener"] = cl_results
               logger.info("Phase B++++: CourtListener returned %d cases", len(cl_results.get("cases", [])))
       except Exception:
           logger.warning("Phase B++++: CourtListener search failed (non-blocking)", exc_info=True)

       # Phase B+++++: FMP institutional ownership + estimates (non-blocking).
       try:
           fmp_inst = self._fmp_client.fetch_institutional_ownership(state.ticker, cache=self._cache)
           if fmp_inst:
               acquired.market_data["fmp_institutional"] = fmp_inst
           fmp_est = self._fmp_client.fetch_analyst_estimates(state.ticker, cache=self._cache)
           if fmp_est:
               acquired.market_data["fmp_estimates"] = fmp_est
           logger.info("Phase B+++++: FMP enrichment complete")
       except Exception:
           logger.warning("Phase B+++++: FMP enrichment failed (non-blocking)", exc_info=True)
       ```
