```python
def extract_business_model(state: AnalysisState) -> dict[str, Any]:
    """Extract business model profile data for template."""
    prof = state.company
    if prof is None:
        return {}

    # Revenue model type (BMOD-01)
    rev_model = sv_val(prof.revenue_model_type, None) if prof.revenue_model_type else None

    # Concentration risk (BMOD-02) -- already in extract_company() segments/geo/customers
    # but here we provide the composite assessment
    concentration_score = 0
    concentration_flags: list[str] = []
    if prof.revenue_segments:
        max_seg = max(
            (float(s.value.get("percentage", s.value.get("pct", 0)) or 0) for s in prof.revenue_segments),
            default=0,
        )
        if max_seg > 50:
            concentration_score += 1
            concentration_flags.append(f"Single segment {max_seg:.0f}%")
    if prof.customer_concentration:
        max_cust = max(
            (float(c.value.get("revenue_pct", 0) or 0) for c in prof.customer_concentration),
            default=0,
        )
        if max_cust > 10:
            concentration_score += 1
            concentration_flags.append(f"Single customer {max_cust:.0f}%")
    if prof.geographic_footprint:
        max_geo = max(
            (float(g.value.get("percentage", g.value.get("pct", 0)) or 0) for g in prof.geographic_footprint),
            default=0,
        )
        if max_geo > 40:
            concentration_score += 1
            concentration_flags.append(f"Single geography {max_geo:.0f}%")

    concentration_level = "HIGH" if concentration_score >= 3 else "MODERATE" if concentration_score >= 2 else "LOW" if concentration_score >= 1 else "NONE"

    # Key person (BMOD-03)
    key_person: dict[str, Any] | None = None
    if prof.key_person_risk is not None:
        kp = prof.key_person_risk.value
        key_person = {
            "is_founder_led": kp.get("is_founder_led", False),
            "ceo_tenure_years": kp.get("ceo_tenure_years"),
            "has_succession_plan": kp.get("has_succession_plan"),
            "risk_score": kp.get("risk_score", 0),
            "risk_level": "HIGH" if kp.get("risk_score", 0) >= 3 else "MODERATE" if kp.get("risk_score", 0) >= 2 else "LOW",
        }

    # Lifecycle (BMOD-04)
    lifecycle: list[dict[str, str]] = []
    for sv_lc in (prof.segment_lifecycle or []):
        lc = sv_lc.value
        lifecycle.append({
            "name": str(lc.get("name", "Unknown")),
            "stage": str(lc.get("stage", "Unknown")),
            "growth_rate": format_percentage(float(lc.get("growth_rate", 0))) if lc.get("growth_rate") is not None else "N/A",
        })

    # Disruption (BMOD-05)
    disruption: dict[str, Any] | None = None
    if prof.disruption_risk is not None:
        dr = prof.disruption_risk.value
        disruption = {
            "level": dr.get("level", "LOW"),
            "threats": dr.get("threats", []),
            "threat_count": dr.get("threat_count", 0),
        }

    # Segment margins (BMOD-06)
    margins: list[dict[str, str]] = []
    for sv_m in (prof.segment_margins or []):
        m = sv_m.value
        margins.append({
            "name": str(m.get("name", "Unknown")),
            "margin": format_percentage(float(m.get("margin_pct", 0))) if m.get("margin_pct") is not None else "N/A",
            "prior_margin": format_percentage(float(m.get("prior_margin_pct", 0))) if m.get("prior_margin_pct") is not None else "N/A",
            "change_bps": str(int(float(m.get("change_bps", 0)))) if m.get("change_bps") is not None else "N/A",
        })

    return {
        "revenue_model_type": rev_model,
        "concentration_score": concentration_score,
        "concentration_level": concentration_level,
        "concentration_flags": concentration_flags,
        "key_person": key_person,
        "lifecycle": lifecycle,
        "disruption": disruption,
        "segment_margins": margins,
    }
```
