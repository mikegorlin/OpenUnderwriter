    **Step 1: Create `src/do_uw/brain/brain_facet_schema.py`**

    New file — do not edit brain_check_schema.py for this. Create a standalone module:

    ```python
    """FacetSpec schema for brain/facets/*.yaml files.

    A Facet is a composed display unit that declares which Signals (checks) it aggregates
    and how to render them together. Facets are the bridge between atomic Signal knowledge
    (brain/checks/**/*.yaml) and the rendering layer.

    Phase 48 scope: Define the schema and load two representative Facets.
    Phase 49 scope: Full rendering migration where every section reads from Facets.
    """
    from __future__ import annotations

    from pathlib import Path
    from typing import Any

    import yaml
    from pydantic import BaseModel, ConfigDict, Field


    VALID_DISPLAY_TYPES = frozenset(
        {"table", "scorecard_table", "flag_list", "metric_table", "profile_card", "audit_table"}
    )


    class FacetSpec(BaseModel):
        """Schema for a single brain/facets/*.yaml file."""

        id: str = Field(description="Unique Facet identifier (e.g. 'governance', 'red_flags')")
        name: str = Field(description="Human-readable Facet name (e.g. 'Governance Assessment')")
        display_type: str = Field(
            description=(
                "How to render this Facet. Valid values: "
                + ", ".join(sorted(VALID_DISPLAY_TYPES))
            )
        )
        signals: list[str] = Field(
            default_factory=list,
            description=(
                "Ordered list of Signal IDs (check IDs) this Facet aggregates. "
                "For red_flags Facet, signals is empty — triggered flags are sourced "
                "from scoring results at render time."
            ),
        )
        display_config: dict[str, Any] = Field(
            default_factory=dict,
            description="Facet-level rendering overrides (column order, sort, visibility flags, etc.)",
        )

        model_config = ConfigDict(extra="allow")

        def validate_display_type(self) -> None:
            """Raise ValueError if display_type is not a recognised value."""
            if self.display_type not in VALID_DISPLAY_TYPES:
                raise ValueError(
                    f"FacetSpec '{self.id}': unknown display_type '{self.display_type}'. "
                    f"Valid: {sorted(VALID_DISPLAY_TYPES)}"
                )


    def load_facet(path: Path) -> FacetSpec:
        """Load and validate a single brain/facets/*.yaml file."""
        raw = yaml.safe_load(path.read_text(encoding="utf-8"))
        facet = FacetSpec.model_validate(raw)
        facet.validate_display_type()
        return facet


    def load_all_facets(facets_dir: Path) -> dict[str, FacetSpec]:
        """Load all *.yaml files from brain/facets/ and return {facet_id: FacetSpec}."""
        facets: dict[str, FacetSpec] = {}
        if not facets_dir.exists():
            return facets
        for yaml_path in sorted(facets_dir.glob("*.yaml")):
            facet = load_facet(yaml_path)
            facets[facet.id] = facet
        return facets
    ```
