```python
"""Gap search result applicator for the ANALYZE stage.

Applies AcquiredData.brain_targeted_search results to SKIPPED CheckResults.
Called after execute_checks() in AnalyzeStage.run() so that check_results
reflects web evidence where structured sources had no data.

Stage boundary: gap search STORES results in ACQUIRE (brain_targeted_search).
Re-evaluation APPLIES them in ANALYZE (check_results update).
"""
from __future__ import annotations
import logging
from typing import Any

logger = logging.getLogger(__name__)
```

**Function to implement:**

`apply_gap_search_results(acquired_data, analysis_results) -> dict[str, int]`:
- Parameters:
  - `acquired_data`: `AcquiredData` instance (has `brain_targeted_search`)
  - `analysis_results`: `AnalysisResults` instance (has `check_results` dict)
- Return: summary dict `{"updated": int, "triggered": int, "clear": int}`
- Logic:
  ```python
  gap_results = acquired_data.brain_targeted_search
  if not gap_results:
      return {"updated": 0, "triggered": 0, "clear": 0}

  updated = triggered = clear = 0
  for check_id, gap_result in gap_results.items():
      check_result = analysis_results.check_results.get(check_id)
      if check_result is None:
          continue  # Check not in results (shouldn't happen but guard it)
      if check_result.get("status") != "SKIPPED":
          continue  # Only update SKIPPED checks — never overwrite real data

      new_status = gap_result.get("suggested_status")
      if new_status not in ("TRIGGERED", "CLEAR"):
          continue  # No valid suggestion — leave as SKIPPED

      domain = gap_result.get("domain", "web")
      source_str = f"WEB (gap): {domain}" if domain else "WEB (gap)"

      check_result["status"] = new_status
      check_result["confidence"] = "LOW"
      check_result["source"] = source_str
      check_result["data_status"] = "EVALUATED"
      check_result["data_status_reason"] = "Resolved via gap search"
      check_result["evidence"] = (
          f"Gap search: keyword match={gap_result.get('keywords_matched', False)}, "
          f"query='{gap_result.get('query', '')}'"
      )

      updated += 1
      if new_status == "TRIGGERED":
          triggered += 1
      else:
          clear += 1

  if updated > 0:
      logger.info(
          "Gap re-evaluation: %d checks updated (%d TRIGGERED, %d CLEAR)",
          updated, triggered, clear,
      )

  return {"updated": updated, "triggered": triggered, "clear": clear}
  ```

Also add `__all__ = ["apply_gap_search_results"]`.

Note: `check_result` is a `dict` (from `r.model_dump()` in AnalyzeStage). Mutating it mutates the dict in `analysis_results.check_results` directly (dict is mutable). This is safe — the dict was constructed in `AnalyzeStage.run()` immediately before this call.

```python
# Apply brain_targeted_search results to SKIPPED checks.
# Must run BEFORE _run_analytical_engines so temporal/forensic engines

if state.acquired_data is not None:
    try:
        from do_uw.stages.analyze.gap_revaluator import apply_gap_search_results
        gap_summary = apply_gap_search_results(
            state.acquired_data, state.analysis
        )
        if gap_summary["updated"] > 0:
            # Recompute aggregate counts after gap re-evaluation
            # (count statuses directly since we modified check_results in-place)
            triggered_now = sum(
                1 for r in state.analysis.check_results.values()
                if r.get("status") == "TRIGGERED"
            )
            clear_now = sum(
                1 for r in state.analysis.check_results.values()
                if r.get("status") == "CLEAR"

            skipped_now = sum(
                1 for r in state.analysis.check_results.values()
                if r.get("status") == "SKIPPED"
            )

            state.analysis.checks_passed = clear_now
            state.analysis.checks_skipped = skipped_now

                "Post-gap counts: %d triggered, %d clear, %d skipped",
                triggered_now, clear_now, skipped_now,
            )
        # Store gap summary for QA audit template (always set, even if updated=0)
        state.analysis.gap_search_summary = gap_summary
    except Exception:
```
