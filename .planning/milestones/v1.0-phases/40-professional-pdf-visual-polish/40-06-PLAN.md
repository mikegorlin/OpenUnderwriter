     Add LLM prompt header stripping BEFORE the existing markdown stripping:
     ```python
     def _strip_markdown(text: str) -> str:
         if not text:
             return ""
         # Strip LLM prompt headers that leak into rendered output
         # Pattern: "AI Assessment: # COMPANY — NARRATIVE" etc.
         text = re.sub(
             r"^(?:AI\s+Assessment:\s*)?#\s+.*?(?:Assessment|Analysis|Summary|Overview|Risk|Narrative).*$",
             "", text, count=1, flags=re.MULTILINE | re.IGNORECASE,
         )
         # Strip "# AI_RISK —" style headers
         text = re.sub(r"^#\s+AI_\w+\s*[—–-].*$", "", text, count=1, flags=re.MULTILINE)
         # Remove bold markers: **text** -> text
         text = re.sub(r"\*\*(.+?)\*\*", r"\1", text)
         # Remove italic markers: *text* -> text
         text = re.sub(r"\*(.+?)\*", r"\1", text)
         # Remove markdown headers: ## text -> text
         text = re.sub(r"^#{1,6}\s+", "", text, flags=re.MULTILINE)
         return text.strip()
     ```

  2. Add `humanize_theory()` filter to formatters.py:
     Theory codes like A_DISCLOSURE, B_FIDUCIARY, SECT7-F7 appear in
     executive.html.j2 lines 83 and 109 (`finding.theory`, `pos.theory`).
     ```python
     _THEORY_DISPLAY: dict[str, str] = {
         "A_DISCLOSURE": "Securities Disclosure (10b-5)",
         "A_MISREPRESENTATION": "Material Misrepresentation",
         "A_OMISSION": "Material Omission",
         "B_FIDUCIARY": "Fiduciary Duty Breach",
         "B_WASTE": "Corporate Waste",
         "B_LOYALTY": "Duty of Loyalty",
         "C_REGULATORY": "Regulatory Non-Compliance",
         "C_ENFORCEMENT": "Enforcement Action",
         "D_EMPLOYMENT": "Employment Practices",
         "D_ERISA": "ERISA Violation",
         "E_ANTITRUST": "Antitrust / Competition",
         "E_IP": "Intellectual Property",
     }

     def humanize_theory(code: str) -> str:
         """Convert internal theory codes to human-readable labels.

         Handles patterns like:
         - A_DISCLOSURE -> "Securities Disclosure (10b-5)"
         - SECT7-F7 -> "Factor 7"
         - Unknown codes -> title-cased with underscores as spaces
         """
         if not code:
             return ""
         # Check direct mapping first
         if code in _THEORY_DISPLAY:
             return _THEORY_DISPLAY[code]
         # SECT7-FN pattern
         m = re.match(r"SECT\d+-F(\d+)", code)
         if m:
             return f"Factor {m.group(1)}"
         # Fallback: humanize
         return code.replace("_", " ").title()
     ```
