```python
import hashlib

def _compute_chart_data_hash(state: AnalysisState) -> str:
    """Hash the price data inputs to detect changes."""
    md = state.acquired_data.market_data if state.acquired_data else {}
    # Hash the raw price history keys that drive chart rendering
    keys = ["history_1y", "history_5y", "sector_history_1y", "sector_history_5y",
            "spy_history_1y", "spy_history_5y"]
    hasher = hashlib.sha256()
    for k in keys:
        val = md.get(k, {})
        # Hash Close prices (the data that matters for chart rendering)
        closes = val.get("Close", []) if isinstance(val, dict) else []
        hasher.update(f"{k}:{len(closes)}:{closes[:3]}:{closes[-3:] if closes else []}".encode())
    return hasher.hexdigest()[:16]

def _generate_chart_images(
    state: AnalysisState, chart_dir: Path, ds: DesignSystem,
) -> None:
    """Generate all chart PNGs and save to chart_dir.

    Implements data-hash caching: computes a hash of the price data inputs
    and writes it to chart_dir/.data_hash. If the hash matches on next run,
    skips chart regeneration entirely.

    Creates: stock_1y.png, stock_5y.png, radar.png.
    Each chart is generated and saved independently -- individual failures
    are logged but don't block other charts.
    """
    # Data-hash cache guard: skip if price data unchanged
    hash_file = chart_dir / ".data_hash"
    current_hash = _compute_chart_data_hash(state)
    if hash_file.exists():
        existing_hash = hash_file.read_text().strip()
        stock_1y_exists = (chart_dir / "stock_1y.png").exists()
        stock_5y_exists = (chart_dir / "stock_5y.png").exists()
        if existing_hash == current_hash and (stock_1y_exists or stock_5y_exists):
            logger.info("Chart data unchanged (hash=%s), skipping regeneration", current_hash)
            return

    from do_uw.stages.render.charts.stock_charts import create_stock_chart

    # Stock performance charts
    for period, filename in [("1Y", "stock_1y.png"), ("5Y", "stock_5y.png")]:
        try:
            buf = create_stock_chart(state, period=period, ds=ds)
            if buf is not None:
                (chart_dir / filename).write_bytes(buf.read())
                buf.seek(0)  # Reset for potential reuse
                logger.info("Generated chart: %s", filename)
            else:
                logger.info("No data for chart: %s", filename)
        except Exception:
            logger.exception("Failed to generate chart: %s", filename)

    # Radar chart (existing)
    try:
        from do_uw.stages.render.chart_helpers import create_radar_chart
        if state.analysis and state.analysis.scoring:
            scoring = state.analysis.scoring
            if scoring.factor_scores:
                categories = [f.factor_name for f in scoring.factor_scores]
                values = [f.weighted_score for f in scoring.factor_scores]
                max_val = max(values) * 1.2 if values else 10.0
                buf = create_radar_chart(categories, values, max_val, "10-Factor Risk Profile", ds)
                (chart_dir / "radar.png").write_bytes(buf.read())
                logger.info("Generated chart: radar.png")
    except Exception:
        logger.exception("Failed to generate radar chart")

    # Write hash file after successful generation
    hash_file.write_text(current_hash)
```
