    ```python
    def _add_llm_report(
        reports: list[ExtractionReport],
        extractor_name: str,
        source: str,
        model: BaseModel,
    ) -> None:
        """Create extraction report from LLM-populated model."""
        # Count non-None fields as "found"
        found = [
            name for name, field_info in model.model_fields.items()
            if getattr(model, name, None) is not None
        ]
        reports.append(ExtractionReport(
            extractor_name=extractor_name,
            expected_fields=list(model.model_fields.keys()),
            found_fields=found,
            source_filing=source,
        ))
    ```
