    ```python
    """Shared helpers for deserializing LLM extraction results from state.

    Pre-deserializes LLM results once at sub-orchestrator level so individual
    converter functions receive typed Pydantic models, not raw dicts.
    """
    from __future__ import annotations

    import logging
    from typing import Any

    from do_uw.models.state import AnalysisState
    from do_uw.stages.extract.llm.schemas import DEF14AExtraction, TenKExtraction

    logger = logging.getLogger(__name__)


    def get_llm_def14a(state: AnalysisState) -> DEF14AExtraction | None:
        """Deserialize LLM DEF 14A extraction from state.

        Searches llm_extractions for a key starting with 'DEF 14A:'.
        Returns the first match deserialized into DEF14AExtraction,
        or None if no DEF 14A extraction exists or deserialization fails.
        """
        return _get_llm_extraction(state, "DEF 14A:", DEF14AExtraction)


    def get_llm_ten_k(state: AnalysisState) -> TenKExtraction | None:
        """Deserialize LLM 10-K extraction from state.

        Searches for keys starting with '10-K:' or '20-F:' (FPI equivalent).
        Returns the first match deserialized into TenKExtraction,
        or None if no 10-K extraction exists or deserialization fails.
        """
        result = _get_llm_extraction(state, "10-K:", TenKExtraction)
        if result is not None:
            return result
        return _get_llm_extraction(state, "20-F:", TenKExtraction)


    def _get_llm_extraction[T](
        state: AnalysisState,
        key_prefix: str,
        model_class: type[T],
    ) -> T | None:
        """Generic LLM extraction deserializer."""
        if state.acquired_data is None:
            return None
        extractions = state.acquired_data.llm_extractions
        if not extractions:
            return None
        for key, data in extractions.items():
            if key.startswith(key_prefix):
                try:
                    if isinstance(data, dict):
                        # model_validate is on BaseModel -- need cast for generic
                        from pydantic import BaseModel
                        if issubclass(model_class, BaseModel):
                            return model_class.model_validate(data)  # type: ignore[union-attr]
                    return None
                except Exception:
                    logger.warning(
                        "Failed to deserialize LLM extraction for %s",
                        key,
                        exc_info=True,
                    )
                    return None
        return None
    ```

