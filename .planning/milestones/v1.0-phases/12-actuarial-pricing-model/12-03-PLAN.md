```python
    def _enrich_actuarial_pricing(
        self,
        state: AnalysisState,
        scoring: Any,
        inherent_risk: InherentRiskBaseline,
    ) -> None:
        """Optionally enrich state with actuarial pricing model output.

        Non-breaking: failures are logged and silently ignored.
        Actuarial pricing is additive -- its absence never blocks the pipeline.
        """
        try:
            from do_uw.stages.score.actuarial_layer_pricing import (
                build_actuarial_pricing,
            )

            # Filing probability from inherent risk (computed in Step 4)
            filing_probability_pct = inherent_risk.company_adjusted_rate_pct

            # Severity scenarios from SCORE stage
            severity = scoring.severity_scenarios

            # Determine case type (default for now; could be refined based on
            # active litigation type in future phases)
            case_type = "default"

            # Sector and market cap tier for ILF alpha selection
            sector: str | None = None
            if (
                state.company is not None
                and state.company.identity.sector is not None
            ):
                sector = str(state.company.identity.sector.value)
            market_cap_tier = inherent_risk.market_cap_tier

            # Market position for calibration (may not exist)
            market_position = None
            if (
                state.executive_summary is not None
                and state.executive_summary.deal_context.market_intelligence is not None
                and state.executive_summary.deal_context.market_intelligence.has_data
            ):
                # Build a lightweight MarketPosition-like object from MarketIntelligence
                # to avoid importing pricing_analytics at runtime
                mi = state.executive_summary.deal_context.market_intelligence
                market_position = _MarketPositionProxy(
                    peer_count=mi.peer_count,
                    confidence_level=mi.confidence_level,
                    median_rate_on_line=mi.median_rate_on_line,
                )

            # Load actuarial config
            import json
            from pathlib import Path
            config_path = Path(__file__).parent.parent.parent / "config" / "actuarial.json"
            actuarial_config = json.loads(config_path.read_text())

            result = build_actuarial_pricing(
                filing_probability_pct=filing_probability_pct,
                severity_scenarios=severity,
                case_type=case_type,
                sector=sector,
                market_cap_tier=market_cap_tier,
                market_position=market_position,
                actuarial_config=actuarial_config,
            )

            scoring.actuarial_pricing = result

            if result.has_data:
                logger.info(
                    "Actuarial pricing: expected_loss=$%.0f, "
                    "total_premium=$%.0f, layers=%d",
                    result.expected_loss.total_expected_loss if result.expected_loss else 0,
                    result.total_indicated_premium,
                    len(result.layer_pricing),
                )
            else:
                logger.info("Actuarial pricing: insufficient data")

        except Exception:
            logger.debug(
                "Actuarial pricing unavailable for %s",
                state.ticker,
                exc_info=True,
            )
```
