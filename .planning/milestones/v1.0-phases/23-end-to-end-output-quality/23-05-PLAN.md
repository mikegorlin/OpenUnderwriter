```python
# ---------------------------------------------------------------------------
# .docx output validation helpers
# ---------------------------------------------------------------------------

def load_docx(ticker: str) -> Any | None:
    """Load the generated .docx worksheet for a ticker."""
    from docx import Document  # type: ignore[import-untyped]
    docx_path = OUTPUT_DIR / ticker / f"{ticker}_worksheet.docx"
    if not docx_path.exists():
        return None
    return Document(str(docx_path))


def read_docx_tables(doc: Any) -> list[list[list[str]]]:
    """Read all tables from a docx Document as nested string lists."""
    tables: list[list[list[str]]] = []
    for table in doc.tables:
        rows: list[list[str]] = []
        for row in table.rows:
            cells = [cell.text.strip() for cell in row.cells]
            rows.append(cells)
        tables.append(rows)
    return tables


def read_docx_text(doc: Any) -> str:
    """Read all paragraph text from a docx Document."""
    return "\n".join(p.text for p in doc.paragraphs)


def find_in_tables(
    tables: list[list[list[str]]],
    label: str,
) -> list[tuple[int, int, list[str]]]:
    """Find all table rows where the first cell contains label (case-insensitive)."""
    results: list[tuple[int, int, list[str]]] = []
    for ti, table in enumerate(tables):
        for ri, row in enumerate(table):
            if row and label.lower() in row[0].lower():
                results.append((ti, ri, row))
    return results


def find_text_containing(doc: Any, substring: str) -> list[str]:
    """Find all paragraphs containing a substring (case-insensitive)."""
    results: list[str] = []
    for p in doc.paragraphs:
        if substring.lower() in p.text.lower():
            results.append(p.text)
    return results
```

```python
"""Process validation harness for generated .docx worksheets.

Reads the final rendered document (not state.json) and validates
that factual claims are correct. Each test class covers one ticker.

Tests are skipped when the .docx file is not present (requires
a prior pipeline run: `do-uw analyze {TICKER}`).
"""

import re
import pytest
from tests.ground_truth.helpers import (
    load_docx,
    read_docx_tables,
    read_docx_text,
    find_in_tables,
    find_text_containing,
)
from tests.ground_truth.xom import GROUND_TRUTH as XOM_TRUTH
from tests.ground_truth.smci import GROUND_TRUTH as SMCI_TRUTH


def _get_doc(ticker: str):
    """Load .docx or skip test."""
    doc = load_docx(ticker)
    if doc is None:
        pytest.skip(f"No .docx found for {ticker} (run pipeline first)")
    return doc


class TestXOMOutput:
    """Validate XOM worksheet against known facts."""

    def test_employee_count_reasonable(self):
        doc = _get_doc("XOM")
        text = read_docx_text(doc)
        tables = read_docx_tables(doc)
        facts = XOM_TRUTH.get("output_facts", {})
        # Find employee count in tables or text
        emp_rows = find_in_tables(tables, "employee")
        if emp_rows:
            # Parse number from the value cell
            _, _, row = emp_rows[0]
            if len(row) >= 2:
                num_str = re.sub(r'[^\d]', '', row[1])
                if num_str:
                    emp_num = int(num_str)
                    assert emp_num >= facts.get("employee_count_min", 50000), (
                        f"XOM employees {emp_num} < {facts.get('employee_count_min')}"
                    )

    def test_sector_correct(self):
        doc = _get_doc("XOM")
        text = read_docx_text(doc)
        assert "Energy" in text, "XOM should be classified as Energy sector"

    def test_shares_no_dollar_prefix(self):
        doc = _get_doc("XOM")
        tables = read_docx_tables(doc)
        share_rows = find_in_tables(tables, "shares outstanding")
        for _, _, row in share_rows:
            for cell in row[1:]:
                assert not cell.startswith("$"), (
                    f"Shares Outstanding has $ prefix: {cell}"
                )

    def test_auditor_present(self):
        doc = _get_doc("XOM")
        text = read_docx_text(doc)
        # PwC or PricewaterhouseCoopers should appear
        assert any(
            kw in text for kw in [
                "PricewaterhouseCoopers", "PwC", "Pricewaterhouse",
            ]
        ), "XOM auditor (PwC) not found in worksheet"


class TestSMCIOutput:
    """Validate SMCI worksheet against known facts."""

    def test_sector_tech(self):
        doc = _get_doc("SMCI")
        text = read_docx_text(doc)
        assert "Technology" in text, "SMCI should be classified as Technology sector"
        assert "Industrials" not in text or "Technology" in text

    def test_employee_count_reasonable(self):
        doc = _get_doc("SMCI")
        tables = read_docx_tables(doc)
        emp_rows = find_in_tables(tables, "employee")
        if emp_rows:
            _, _, row = emp_rows[0]
            if len(row) >= 2:
                num_str = re.sub(r'[^\d]', '', row[1])
                if num_str:
                    emp_num = int(num_str)
                    assert emp_num >= 1000, f"SMCI employees {emp_num} too low"

    def test_has_data_quality_notice_or_known_events(self):
        """SMCI worksheet should either have blind spot findings OR a data quality notice."""
        doc = _get_doc("SMCI")
        text = read_docx_text(doc)
        has_blind_spot_notice = "blind spot detection" in text.lower()
        has_hindenburg = "hindenburg" in text.lower()
        has_auditor_issue = any(
            kw in text.lower() for kw in ["ey resign", "ernst & young resign", "auditor resign"]
        )
        assert has_blind_spot_notice or has_hindenburg or has_auditor_issue, (
            "SMCI worksheet should have either blind spot detection notice "
            "or findings about Hindenburg/auditor resignation"
        )
```
