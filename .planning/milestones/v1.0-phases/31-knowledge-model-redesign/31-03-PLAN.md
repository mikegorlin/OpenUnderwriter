```python
def narrow_result(
    check_id: str,
    data: dict[str, Any],
    check_def: dict[str, Any] | None = None,
) -> dict[str, Any]:
    """Narrow a mapper result dict to the field relevant to this check.

    Resolution order:
    1. data_strategy.field_key from check definition (Phase 31 declarative)
    2. Exact check_id match in FIELD_FOR_CHECK (legacy)
    3. Fallback: return full data dict (existing behavior)
    """
    # Phase 31: declarative field_key from check definition
    if check_def is not None:
        ds = check_def.get("data_strategy")
        if isinstance(ds, dict):
            fk = ds.get("field_key")
            if fk is not None:
                if fk in data:
                    return {fk: data[fk]}
                return {}  # Intended field missing -> DATA_UNAVAILABLE

    # Legacy: exact match in FIELD_FOR_CHECK
    field = FIELD_FOR_CHECK.get(check_id)
    if field is not None:
        if field in data:
            return {field: data[field]}
        return {}
    return data
```

```python
_CHECK_META_CACHE: dict[str, dict[str, Any]] | None = None

def _load_check_metadata() -> dict[str, dict[str, Any]]:
    """Load check definitions for gap enrichment. Cached."""
    global _CHECK_META_CACHE
    if _CHECK_META_CACHE is not None:
        return _CHECK_META_CACHE
    try:
        from do_uw.knowledge.compat_loader import BackwardCompatLoader
        loader = BackwardCompatLoader()
        checks_data = loader.load_checks()
        _CHECK_META_CACHE = {
            str(c.get("id", "")): c
            for c in checks_data.get("checks", [])
        }
    except Exception:
        _CHECK_META_CACHE = {}
    return _CHECK_META_CACHE
```
