    ```python
    # LLM AI risk factor supplement
    from do_uw.stages.extract.llm_helpers import get_llm_ten_k
    llm_ten_k = get_llm_ten_k(state)

    if llm_ten_k and llm_ten_k.risk_factors:
        # Find risk factors with AI-related categories
        ai_factors = [
            rf for rf in llm_ten_k.risk_factors
            if rf.category and "ai" in rf.category.lower()
        ]
        if ai_factors:
            # Supplement AI disclosure data with LLM-extracted AI risk factors
            # These provide richer context than keyword counting
            if ai_disclosure is not None:
                # Add AI-specific risk factor descriptions to disclosure
                for af in ai_factors:
                    if af.description and af.description not in ai_disclosure.risk_factors:
                        ai_disclosure.risk_factors.append(af.description)
            logger.info("SECT8: Supplemented AI risk with %d LLM AI factors", len(ai_factors))
    ```

    IMPORTANT: Read extract_ai_risk.py to understand the actual AI risk model and variable names. The `ai_disclosure` model may have different field names. The key principle: LLM risk factors categorized as "AI" supplement existing keyword-based analysis.
