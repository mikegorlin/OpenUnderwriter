    ```python
    # Core types that every model file imports
    class Confidence(str, Enum):
        HIGH = "HIGH"      # Audited/official (SEC filings, exchange data)
        MEDIUM = "MEDIUM"  # Unaudited/estimates (press releases, analyst data)
        LOW = "LOW"        # Derived/web (news, web search, single source)

    T = TypeVar("T")

    class SourcedValue(BaseModel, Generic[T]):
        """Every data point carries provenance. NON-NEGOTIABLE per CLAUDE.md."""
        value: T
        source: str = Field(description="Filing type + date + URL/CIK reference")
        confidence: Confidence
        as_of: datetime = Field(description="When this data was valid")
        retrieved_at: datetime = Field(default_factory=datetime.utcnow)

    class StageStatus(str, Enum):
        PENDING = "pending"
        RUNNING = "running"
        COMPLETED = "completed"
        FAILED = "failed"
        SKIPPED = "skipped"

    class StageResult(BaseModel):
        """Tracks execution status of a single pipeline stage."""
        stage: str
        status: StageStatus = StageStatus.PENDING
        started_at: datetime | None = None
        completed_at: datetime | None = None
        duration_seconds: float | None = None
        error: str | None = None

    class DataFreshness(str, Enum):
        CURRENT = "CURRENT"    # <90 days
        AGING = "AGING"        # 90-180 days
        STALE = "STALE"        # >180 days
    ```

    Use `from __future__ import annotations` in all model files for forward reference support.
    Import `datetime` from the `datetime` module (not `datetime.datetime`).
    Use Pydantic v2 `model_config = ConfigDict(...)` syntax, NOT the old `class Config:` syntax.

    **Step 5: Create ConfigLoader:**
    Create `src/do_uw/config/loader.py`:

    ```python
    class BrainConfig(BaseModel):
        """Loaded and validated brain/ knowledge assets."""
        checks: dict  # Full check registry (validated as dict for Phase 1; typed model in Phase 6)
        scoring: dict  # Consolidated scoring config
        patterns: dict  # 17 composite patterns
        sectors: dict   # Sector baselines
        red_flags: dict # Critical red flags

    class ConfigLoader:
        """Loads and validates all config from brain/ directory."""

        def __init__(self, brain_dir: Path | None = None):
            self.brain_dir = brain_dir or self._default_brain_dir()

        @staticmethod
        def _default_brain_dir() -> Path:
            """brain/ lives inside the package."""
            return Path(__file__).parent.parent / "brain"

        def load_all(self) -> BrainConfig:
            """Load all brain config files. Raises FileNotFoundError or ValidationError on failure."""
            return BrainConfig(
                checks=self._load_json("checks.json"),
                scoring=self._load_json("scoring.json"),
                patterns=self._load_json("patterns.json"),
                sectors=self._load_json("sectors.json"),
                red_flags=self._load_json("red_flags.json"),
            )

        def _load_json(self, filename: str) -> dict:
            path = self.brain_dir / filename
            if not path.exists():
                raise FileNotFoundError(f"Brain config not found: {path}")
            return json.loads(path.read_text())

        # Individual loaders for when you only need one
        def load_checks(self) -> dict: ...
        def load_scoring(self) -> dict: ...
        def load_patterns(self) -> dict: ...
        def load_sectors(self) -> dict: ...
        def load_red_flags(self) -> dict: ...
    ```

    Update `src/do_uw/config/__init__.py` to re-export `ConfigLoader` and `BrainConfig`.

    Note: In Phase 1, config validation uses `dict` types. In Phase 6, when the scoring engine is built, these will be replaced with typed Pydantic models for each config type. This is intentional -- define structure now, type deeply later.
