    ```python
    """Pipeline orchestrator. Runs stages sequentially with state persistence."""
    from pathlib import Path
    from do_uw.models.state import AnalysisState, PIPELINE_STAGES
    from do_uw.models.common import StageStatus
    from do_uw.stages.resolve import ResolveStage
    from do_uw.stages.acquire import AcquireStage
    # ... all 7 stage imports

    class Pipeline:
        """Orchestrates the 7-stage D&O analysis pipeline.

        Features:
        - Sequential execution with validation gates
        - State persistence to JSON after each stage
        - Resume-from-failure: skips COMPLETED stages on re-run
        """

        def __init__(self, output_dir: Path | None = None):
            self.output_dir = output_dir or Path.cwd() / ".do-uw"
            self.output_dir.mkdir(parents=True, exist_ok=True)
            self.stages = [
                ResolveStage(),
                AcquireStage(),
                ExtractStage(),
                AnalyzeStage(),
                ScoreStage(),
                BenchmarkStage(),
                RenderStage(),
            ]

        def run(self, state: AnalysisState, on_stage_start=None, on_stage_complete=None) -> AnalysisState:
            """Run all stages sequentially. Skips COMPLETED stages (resume support).

            Args:
                state: The analysis state to process
                on_stage_start: Optional callback(stage_name) called before each stage
                on_stage_complete: Optional callback(stage_name, status) called after each stage
            """
            for stage in self.stages:
                stage_result = state.stages[stage.name]

                # Resume support: skip completed stages
                if stage_result.status == StageStatus.COMPLETED:
                    continue

                # Validate input
                errors = stage.validate_input(state)
                if errors:
                    stage_result.status = StageStatus.FAILED
                    stage_result.error = "; ".join(errors)
                    self._save_state(state)
                    raise ValueError(f"Stage {stage.name} validation failed: {stage_result.error}")

                # Notify start
                if on_stage_start:
                    on_stage_start(stage.name)

                # Execute
                try:
                    state = stage.run(state)
                except Exception as e:
                    stage_result.status = StageStatus.FAILED
                    stage_result.error = str(e)
                    self._save_state(state)
                    raise

                # Notify complete
                if on_stage_complete:
                    on_stage_complete(stage.name, stage_result.status)

                # Persist state after each stage
                self._save_state(state)

            return state

        def _save_state(self, state: AnalysisState) -> Path:
            """Save state to JSON file."""
            state_path = self.output_dir / f"{state.ticker.lower()}_state.json"
            state_path.write_text(state.model_dump_json(indent=2))
            return state_path

        def load_state(self, ticker: str) -> AnalysisState | None:
            """Load existing state for resume. Returns None if no prior state exists."""
            state_path = self.output_dir / f"{ticker.lower()}_state.json"
            if state_path.exists():
                return AnalysisState.model_validate_json(state_path.read_text())
            return None
    ```

    The `on_stage_start` and `on_stage_complete` callbacks let the CLI display progress without the pipeline knowing about Rich.

    **Step 1: Create `src/do_uw/cache/sqlite_cache.py`:**

    ```python
    """SQLite cache for API responses and analysis results. ARCH-04."""
    import sqlite3
    from pathlib import Path
    from datetime import datetime, timedelta, timezone

    DEFAULT_CACHE_DIR = Path.home() / ".do-uw" / "cache"

    class AnalysisCache:
        """Key-value cache backed by SQLite.

        - Initializes database on first use (creates tables if not exist)
        - TTL-based expiration
        - Thread-safe (SQLite handles locking)
        """

        def __init__(self, db_path: Path | None = None):
            self.db_path = db_path or DEFAULT_CACHE_DIR / "cache.db"
            self.db_path.parent.mkdir(parents=True, exist_ok=True)
            self._init_db()

        def _init_db(self) -> None:
            """Create tables if they don't exist."""
            with sqlite3.connect(self.db_path) as conn:
                conn.execute("""
                    CREATE TABLE IF NOT EXISTS cache (
                        key TEXT PRIMARY KEY,
                        value TEXT NOT NULL,
                        source TEXT NOT NULL,
                        created_at TEXT NOT NULL,
                        expires_at TEXT,
                        content_type TEXT DEFAULT 'json'
                    )
                """)
                conn.execute("""
                    CREATE TABLE IF NOT EXISTS schema_version (
                        version INTEGER PRIMARY KEY
                    )
                """)
                if not conn.execute("SELECT version FROM schema_version").fetchone():
                    conn.execute("INSERT INTO schema_version (version) VALUES (1)")

        def get(self, key: str) -> str | None:
            """Get cached value by key. Returns None if missing or expired."""
            with sqlite3.connect(self.db_path) as conn:
                row = conn.execute(
                    "SELECT value, expires_at FROM cache WHERE key = ?", (key,)
                ).fetchone()
                if row is None:
                    return None
                value, expires_at = row
                if expires_at:
                    expiry = datetime.fromisoformat(expires_at)
                    if expiry < datetime.now(timezone.utc):
                        conn.execute("DELETE FROM cache WHERE key = ?", (key,))
                        return None
                return value

        def set(self, key: str, value: str, source: str, ttl_hours: int = 24) -> None:
            """Set cached value with TTL."""
            now = datetime.now(timezone.utc)
            expires_at = (now + timedelta(hours=ttl_hours)).isoformat()
            with sqlite3.connect(self.db_path) as conn:
                conn.execute(
                    """INSERT OR REPLACE INTO cache
                       (key, value, source, created_at, expires_at)
                       VALUES (?, ?, ?, ?, ?)""",
                    (key, value, source, now.isoformat(), expires_at),
                )

        def clear(self) -> None:
            """Clear all cached entries."""
            with sqlite3.connect(self.db_path) as conn:
                conn.execute("DELETE FROM cache")

        def stats(self) -> dict[str, int]:
            """Return cache statistics."""
            with sqlite3.connect(self.db_path) as conn:
                total = conn.execute("SELECT COUNT(*) FROM cache").fetchone()[0]
                expired = conn.execute(
                    "SELECT COUNT(*) FROM cache WHERE expires_at < ?",
                    (datetime.now(timezone.utc).isoformat(),)
                ).fetchone()[0]
                return {"total": total, "active": total - expired, "expired": expired}
    ```

    ```python
    """CLI entry point for do-uw. CORE-03."""
    from pathlib import Path
    import typer
    from rich.console import Console
    from rich.table import Table
    from rich.status import Status

    from do_uw.models.state import AnalysisState, PIPELINE_STAGES
    from do_uw.models.common import StageStatus
    from do_uw.pipeline import Pipeline
    from do_uw.cache import AnalysisCache

    app = typer.Typer(name="do-uw", help="D&O Liability Underwriting Worksheet")
    console = Console()

    STATUS_STYLES = {
        StageStatus.PENDING: "[dim]PENDING[/dim]",
        StageStatus.RUNNING: "[yellow]RUNNING[/yellow]",
        StageStatus.COMPLETED: "[green]COMPLETED[/green]",
        StageStatus.FAILED: "[red]FAILED[/red]",
        StageStatus.SKIPPED: "[dim]SKIPPED[/dim]",
    }

    @app.command()
    def analyze(
        ticker: str = typer.Argument(help="Stock ticker symbol (e.g., AAPL)"),
        verbose: bool = typer.Option(False, "--verbose", "-v", help="Show detailed output"),
        output_dir: Path = typer.Option(
            Path.cwd() / ".do-uw", "--output", "-o", help="Output directory"
        ),
    ) -> None:
        """Analyze a company for D&O underwriting."""
        ticker = ticker.upper()
        console.print(f"\n[bold]D&O Underwriting Analysis: {ticker}[/bold]\n")

        # Initialize cache
        cache = AnalysisCache()
        if verbose:
            stats = cache.stats()
            console.print(f"  Cache: {stats['active']} active entries at {cache.db_path}")

        # Create or resume state
        pipeline = Pipeline(output_dir=output_dir)
        state = pipeline.load_state(ticker)
        if state:
            console.print("[dim]Resuming from previous state...[/dim]")
        else:
            state = AnalysisState(ticker=ticker)

        # Show initial stage status
        _print_stages(state)

        # Run pipeline with progress callbacks
        def on_stage_start(name: str) -> None:
            console.print(f"\n  [yellow]>>> Running {name.upper()}...[/yellow]")

        def on_stage_complete(name: str, status: StageStatus) -> None:
            style = STATUS_STYLES.get(status, str(status))
            console.print(f"  [green]<<< {name.upper()} {style}[/green]")

        try:
            state = pipeline.run(
                state,
                on_stage_start=on_stage_start,
                on_stage_complete=on_stage_complete,
            )
            console.print("\n[bold green]Analysis complete.[/bold green]\n")
            _print_stages(state)
        except Exception as e:
            console.print(f"\n[bold red]Pipeline failed: {e}[/bold red]\n")
            _print_stages(state)
            raise typer.Exit(code=1) from e

    def _print_stages(state: AnalysisState) -> None:
        """Print pipeline stage status table."""
        table = Table(title="Pipeline Stages", show_header=True)
        table.add_column("Stage", style="cyan", width=12)
        table.add_column("Status", width=12)
        table.add_column("Duration", justify="right", width=10)

        for stage_name in PIPELINE_STAGES:
            result = state.stages[stage_name]
            status_text = STATUS_STYLES.get(result.status, str(result.status))
            duration = ""
            if result.duration_seconds is not None:
                duration = f"{result.duration_seconds:.2f}s"
            table.add_row(stage_name.upper(), status_text, duration)

        console.print(table)
    ```
