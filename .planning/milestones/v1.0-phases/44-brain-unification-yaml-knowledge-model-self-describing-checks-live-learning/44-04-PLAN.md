```python
@brain_app.command("validate")
def brain_validate(
    checks_dir: Path = typer.Option(
        Path("src/do_uw/brain/checks"),
        "--checks-dir",
        help="Directory containing checks/**/*.yaml files",
    ),
) -> None:
    """Validate all checks YAML files against the unified schema."""
    import glob, yaml
    from pydantic import ValidationError
    # Import or define a minimal CheckValidator Pydantic model inline
    # Required fields to check: id, name, work_type, layer, acquisition_tier,
    #   required_data, worksheet_section, display_when, provenance
    # Valid values: work_type in {extract, evaluate, infer},
    #   layer in {hazard, signal, peril_confirming},
    #   acquisition_tier in {L1, L2, L3, L4}

    errors: list[str] = []
    warnings: list[str] = []
    total = 0

    pattern = str(checks_dir / "**" / "*.yaml")
    for yaml_file in sorted(glob.glob(pattern, recursive=True)):
        data = yaml.safe_load(Path(yaml_file).read_text())
        checks = data if isinstance(data, list) else data.get("checks", [])
        for check in checks:
            total += 1
            cid = check.get("id", "UNKNOWN")
            # Validate required fields
            for field in ["id", "name", "work_type", "layer", "acquisition_tier",
                          "required_data", "worksheet_section", "display_when", "provenance"]:
                if field not in check or check[field] is None:
                    errors.append(f"{yaml_file}: {cid}: missing required field '{field}'")
            # Validate enum values
            if check.get("work_type") not in ("extract", "evaluate", "infer", None):
                errors.append(f"{yaml_file}: {cid}: invalid work_type '{check.get('work_type')}'")
            if check.get("layer") not in ("hazard", "signal", "peril_confirming", None):
                errors.append(f"{yaml_file}: {cid}: invalid layer '{check.get('layer')}'")
            if check.get("acquisition_tier") not in ("L1", "L2", "L3", "L4", None):
                errors.append(f"{yaml_file}: {cid}: invalid acquisition_tier")
            # Warn on deprecated fields still present
            for dep in ("pillar", "category", "signal_type", "hazard_or_signal", "content_type"):
                if dep in check:
                    warnings.append(f"{yaml_file}: {cid}: deprecated field '{dep}' still present")

    # Print summary
    if errors:
        typer.echo(f"VALIDATION FAILED: {len(errors)} errors, {total} checks checked")
        for e in errors[:20]:  # cap output
            typer.echo(f"  ERROR: {e}")
        if len(errors) > 20:
            typer.echo(f"  ... and {len(errors) - 20} more errors")
        raise typer.Exit(1)
    else:
        typer.echo(f"VALIDATION PASSED: {total} checks valid, {len(warnings)} warnings")
        for w in warnings[:10]:
            typer.echo(f"  WARN: {w}")
```

**brain unlinked command:**
```python
@brain_app.command("unlinked")
def brain_unlinked(
    checks_dir: Path = typer.Option(
        Path("src/do_uw/brain/checks"),
        "--checks-dir",
    ),
    domain: Optional[str] = typer.Option(None, "--domain", help="Filter by domain prefix (e.g. GOV, FIN)"),
) -> None:
    """List all checks with unlinked: true (no causal chain assignment)."""
    import glob, yaml

    unlinked_checks: list[dict] = []
    pattern = str(checks_dir / "**" / "*.yaml")
    for yaml_file in sorted(glob.glob(pattern, recursive=True)):
        data = yaml.safe_load(Path(yaml_file).read_text())
        checks = data if isinstance(data, list) else data.get("checks", [])
        for check in checks:
            if check.get("unlinked", False):
                if domain is None or check.get("id", "").startswith(domain):
                    unlinked_checks.append(check)

    typer.echo(f"Unlinked checks: {len(unlinked_checks)}")
    for c in unlinked_checks:
        typer.echo(f"  {c['id']:40s}  {c.get('name', '')[:60]}")
```
