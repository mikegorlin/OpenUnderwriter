**brain add command:**

```python
@brain_app.command("add")
def brain_add(
    domain: str = typer.Option(..., "--domain", help="Domain prefix: biz|fin|gov|exec|lit|stock|fwrd|nlp"),
    source: str = typer.Option(..., "--source", help="URL or citation for the source article/report (REQUIRED)"),
    date: str = typer.Option(..., "--date", help="ISO date of source document YYYY-MM-DD (REQUIRED)"),
    checks_dir: Path = typer.Option(Path("src/do_uw/brain/checks"), "--checks-dir"),
) -> None:
    """Add a new check from an article or research source.

    Opens an interactive YAML template. Validates against schema before saving.
    Runs brain build automatically after saving.
    """
    import yaml, subprocess, re
    from datetime import date as date_type

    # Validate domain
    valid_domains = {"biz", "fin", "gov", "exec", "lit", "stock", "fwrd", "nlp"}
    if domain not in valid_domains:
        typer.echo(f"ERROR: domain must be one of: {', '.join(sorted(valid_domains))}")
        raise typer.Exit(1)

    # Validate date format
    try:
        date_type.fromisoformat(date)
    except ValueError:
        typer.echo(f"ERROR: --date must be ISO format YYYY-MM-DD, got: {date}")
        raise typer.Exit(1)

    # Present the template
    template = f"""# New check from: {source}
# Date: {date}
# Edit ALL fields below, then save and close the editor.
# Required: id, name, work_type, layer, acquisition_tier, required_data, worksheet_section, display_when

id: {domain.upper()}.SUBTYPE.descriptive_name    # e.g. GOV.BOARD.insider_ratio
name: Descriptive check name
work_type: evaluate     # extract | evaluate | infer
layer: signal           # hazard | signal | peril_confirming
acquisition_tier: L2    # L1 (XBRL/filing) | L2 (filing text) | L3 (web/market) | L4 (derived)

# Risk position
factors: []             # e.g. [F9] — scoring factor IDs
peril_ids: []           # empty until causal chain linked
chain_roles: {{}}        # empty until causal chain linked
unlinked: true

# Data acquisition
required_data: []       # e.g. [SEC_PROXY, SEC_10K]
data_locations: {{}}

# Evaluation (for work_type: evaluate)
threshold:
  type: ratio
  red: ""
  yellow: ""
  clear: ""
execution_mode: AUTO
claims_correlation: 0.0
tier: 2
depth: 2

# Presentation
worksheet_section: governance   # company_profile|financial|governance|litigation|scoring|market|forward
display_when: has_data          # always|has_data|fired|critical_only
v6_subsection_ids: []
plaintiff_lenses: [SHAREHOLDERS]

# Provenance (auto-populated from --source and --date flags)
provenance:
  origin: brain_add
  confidence: low
  last_validated: null
  source_url: {source}
  source_date: {date}
  source_author: null
  added_by: null
"""

    # Write to temp file and open in $EDITOR
    import tempfile, os
    with tempfile.NamedTemporaryFile(mode='w', suffix='.yaml', delete=False) as f:
        f.write(template)
        tmp_path = f.name

    editor = os.environ.get("EDITOR", "vi")
    typer.echo(f"Opening editor: {editor} {tmp_path}")
    subprocess.run([editor, tmp_path], check=True)

    # Read back and validate
    try:
        check_data = yaml.safe_load(Path(tmp_path).read_text())
    except yaml.YAMLError as e:
        typer.echo(f"ERROR: Invalid YAML: {e}")
        os.unlink(tmp_path)
        raise typer.Exit(1)
    finally:
        if Path(tmp_path).exists():
            os.unlink(tmp_path)

    # Enforce required fields
    required = ["id", "name", "work_type", "layer", "acquisition_tier",
                "required_data", "worksheet_section", "display_when"]
    missing = [f for f in required if not check_data.get(f)]
    if missing:
        typer.echo(f"ERROR: Missing required fields: {missing}")
        raise typer.Exit(1)

    # Enforce provenance.source_url
    if not check_data.get("provenance", {}).get("source_url"):
        typer.echo("ERROR: provenance.source_url is required for brain add")
        raise typer.Exit(1)

    # Determine target file
    check_id = check_data["id"]
    target_file = _resolve_domain_file(domain, check_id, checks_dir)
    target_file.parent.mkdir(parents=True, exist_ok=True)

    # Append to domain file
    existing = []
    if target_file.exists():
        existing_data = yaml.safe_load(target_file.read_text())
        if isinstance(existing_data, list):
            existing = existing_data
        elif isinstance(existing_data, dict) and "checks" in existing_data:
            existing = existing_data["checks"]

    # Check for duplicate ID
    if any(c.get("id") == check_id for c in existing):
        typer.echo(f"ERROR: Check ID '{check_id}' already exists in {target_file}")
        raise typer.Exit(1)

    existing.append(check_data)
    target_file.write_text(yaml.dump(existing, default_flow_style=False, allow_unicode=True))
    typer.echo(f"Added check '{check_id}' to {target_file}")

    # Run brain build
    typer.echo("Running brain build...")
    result = subprocess.run(["uv", "run", "do-uw", "brain", "build"], capture_output=True, text=True)
    typer.echo(result.stdout)
    if result.returncode != 0:
        typer.echo(f"WARNING: brain build failed: {result.stderr}")
    else:
        typer.echo("brain build complete — new check is active")


def _resolve_domain_file(domain: str, check_id: str, checks_dir: Path) -> Path:
    """Map domain + check_id to the target YAML file path."""
    # For gov/ and fwrd/, use subprefix
    if domain == "gov":
        subprefix = check_id.split(".")[1].lower() if "." in check_id else "effect"
        submap = {"board": "board", "audit": "audit", "exec": "exec_comp",
                  "activist": "activist", "pay": "pay", "rights": "rights"}
        subfile = submap.get(subprefix, "effect")
        return checks_dir / "gov" / f"{subfile}.yaml"
    elif domain == "fwrd":
        subprefix = check_id.split(".")[1].lower() if "." in check_id else "transform"
        submap = {"guid": "guidance", "forecast": "guidance", "spac": "spac",
                  "merger": "spac", "ma": "ma", "acq": "ma"}
        subfile = submap.get(subprefix, "transform")
        return checks_dir / "fwrd" / f"{subfile}.yaml"
    else:
        return checks_dir / f"{domain}.yaml"
```

```python
@brain_app.command("provenance")
def brain_provenance(
    check_id: str = typer.Argument(..., help="Check ID to look up (e.g. GOV.BOARD.independence)"),
) -> None:
    """Show provenance chain for a specific check."""
    import glob, yaml

    checks_dir = Path("src/do_uw/brain/checks")
    found = None
    for yaml_file in glob.glob(str(checks_dir / "**" / "*.yaml"), recursive=True):
        data = yaml.safe_load(Path(yaml_file).read_text())
        checks = data if isinstance(data, list) else data.get("checks", [])
        for check in checks:
            if check.get("id") == check_id:
                found = (check, yaml_file)
                break
        if found:
            break

    if not found:
        typer.echo(f"Check '{check_id}' not found in checks/**/*.yaml")
        raise typer.Exit(1)

    check, yaml_file = found
    prov = check.get("provenance", {})
    typer.echo(f"Check: {check_id}")
    typer.echo(f"File:  {yaml_file}")
    typer.echo(f"Name:  {check.get('name', 'N/A')}")
    typer.echo(f"Provenance:")
    typer.echo(f"  origin:          {prov.get('origin', 'unknown')}")
    typer.echo(f"  confidence:      {prov.get('confidence', 'unknown')}")
    typer.echo(f"  source_url:      {prov.get('source_url', 'null')}")
    typer.echo(f"  source_date:     {prov.get('source_date', 'null')}")
    typer.echo(f"  source_author:   {prov.get('source_author', 'null')}")
    typer.echo(f"  last_validated:  {prov.get('last_validated', 'null')}")
    typer.echo(f"  added_by:        {prov.get('added_by', 'null')}")
    typer.echo(f"Causal chains:    {check.get('chain_roles', {})}")
    typer.echo(f"Peril IDs:        {check.get('peril_ids', [])}")
    typer.echo(f"Unlinked:         {check.get('unlinked', True)}")
```
