    **Step 2: Create `src/do_uw/brain/brain_check_schema.py`**

    ```python
    """Pydantic schema for brain/checks YAML check entries.

    BrainCheckEntry defines the required and optional fields for each check entry
    in brain/checks/**/*.yaml. BrainKnowledgeLoader.load_checks() validates every
    entry against this schema on load, raising RuntimeError on schema violations.

    Derive fields from what is ACTUALLY present in the YAML files (sampled in Phase 45).
    Use Claude's discretion for exact field types based on the audit.
    """
    from __future__ import annotations

    from typing import Any
    from pydantic import BaseModel, Field


    class BrainCheckThreshold(BaseModel):
        """Threshold definition for a check entry."""
        model_config = {"extra": "allow"}  # allow additional threshold keys

        type: str  # "tiered", "boolean", "numeric", etc.
        red: str | None = None
        yellow: str | None = None
        clear: str | None = None


    class BrainCheckProvenance(BaseModel):
        """Provenance metadata for a check entry."""
        model_config = {"extra": "allow"}

        origin: str
        confidence: str | None = None
        last_validated: str | None = None
        source_url: str | None = None
        source_date: str | None = None
        source_author: str | None = None
        added_by: str | None = None


    class BrainCheckEntry(BaseModel):
        """Pydantic model for a single check entry in brain/checks/**/*.yaml.

        Required fields: id, name, work_type, layer, factors, tier, depth, threshold, provenance.
        All other fields are optional with sensible defaults.

        Usage:
            entry = BrainCheckEntry.model_validate(raw_dict)
        """
        model_config = {"extra": "allow"}  # allow unknown fields — don't block load on new fields

        # Required fields
        id: str = Field(..., description="Check identifier (e.g., 'FIN.PROFIT.revenue')")
        name: str = Field(..., description="Human-readable check name")
        work_type: str = Field(..., description="Check work type (evaluate, detect, etc.)")
        layer: str = Field(..., description="Risk framework layer (signal, peril_confirming, etc.)")
        factors: list[str] = Field(..., description="Factor IDs this check contributes to")
        tier: int = Field(..., description="Priority tier (1-3)")
        depth: int = Field(..., description="Analysis depth (1-4)")
        threshold: BrainCheckThreshold = Field(..., description="Threshold definition")
        provenance: BrainCheckProvenance = Field(..., description="Origin and audit metadata")

        # Optional fields with defaults
        peril_ids: list[str] = Field(default_factory=list)
        chain_roles: dict[str, Any] = Field(default_factory=dict)
        unlinked: bool = False
        acquisition_tier: str | None = None
        required_data: list[str] = Field(default_factory=list)
        data_locations: dict[str, Any] = Field(default_factory=dict)
        extraction_hints: dict[str, Any] | None = None
        data_strategy: dict[str, Any] | None = None
        plaintiff_lenses: list[str] = Field(default_factory=list)
        v6_subsection_ids: list[str] = Field(default_factory=list)
        worksheet_section: str | None = None
        display_when: str | None = None
        claims_correlation: float | None = None
        critical_red_flag: bool = False
        sector_adjustments: dict[str, Any] | None = None
    ```

    The load_checks() method returns a dict with a `"checks"` list. After building the checks list (either from BrainDBLoader or from KnowledgeStore fallback), add Pydantic validation for each entry before returning:

    ```python
    from do_uw.brain.brain_check_schema import BrainCheckEntry
    from pydantic import ValidationError

    # In load_checks(), after building the checks list:
    validated: list[dict] = []
    schema_errors: list[str] = []

    for check in result.get("checks", []):
        try:
            BrainCheckEntry.model_validate(check)
            validated.append(check)
        except ValidationError as e:
            check_id = check.get("check_id") or check.get("id", "unknown")
            schema_errors.append(f"  check '{check_id}': {e.error_count()} error(s) — {e.errors()[0]['msg']}")

    if schema_errors:
        error_list = "\n".join(schema_errors[:10])  # cap at 10 to avoid huge error messages
        total = len(schema_errors)
        raise RuntimeError(
            f"BrainKnowledgeLoader.load_checks() found {total} schema violation(s) in brain YAML data.\n"
            f"Fix the YAML entries or update BrainCheckEntry in brain_check_schema.py.\n"
            f"First violations:\n{error_list}"
        )

    result["checks"] = validated
    ```

    Insert this validation block immediately before the `return result` statement in `load_checks()`.
