  **1a. Rewrite ExtractStage orchestrator** (`stages/extract/__init__.py`, ~200 lines):

  Replace the stub with the real implementation. The orchestrator:
  1. Validates preconditions (acquire stage complete, acquired_data exists)
  2. Calls each extractor in dependency order
  3. Collects all ExtractionReports
  4. Produces a consolidated validation summary
  5. Populates state.extracted with all results

  ```python
  class ExtractStage:
      def __init__(self, peers: list[str] | None = None) -> None:
          self._peers = peers

      @property
      def name(self) -> str:
          return "extract"

      def validate_input(self, state: AnalysisState) -> None:
          acquire = state.stages.get("acquire")
          if acquire is None or acquire.status != StageStatus.COMPLETED:
              msg = "Acquire stage must be completed before extract"
              raise ValueError(msg)
          if state.acquired_data is None:
              msg = "No acquired data available for extraction"
              raise ValueError(msg)

      def run(self, state: AnalysisState) -> None:
          state.mark_stage_running(self.name)
          reports: list[ExtractionReport] = []
          try:
              # Phase 1: Company profile (SECT2)
              profile, profile_reports = extract_company_profile(state)
              state.company = profile  # Enriches existing profile
              reports.extend(profile_reports if isinstance(profile_reports, list) else [profile_reports])

              # Phase 2: Financial statements (SECT3-02/03/04)
              statements, stmt_reports = extract_financial_statements(state)
              extracted = _ensure_extracted(state)
              extracted.financials = ExtractedFinancials(statements=statements)
              reports.extend(stmt_reports)

              # Phase 3: Distress indicators (SECT3-07)
              sector = _get_sector(state)
              market_cap = _get_market_cap(state)
              distress, distress_reports = compute_distress_indicators(statements, sector, market_cap)
              extracted.financials.distress = distress
              reports.extend(distress_reports)

              # Phase 4: Earnings quality (SECT3-06)
              eq, eq_report = compute_earnings_quality(statements)
              extracted.financials.earnings_quality = eq
              reports.append(eq_report)

              # Phase 5: Debt analysis (SECT3-08/09/10/11)
              liquidity, leverage, debt_struct, refi, debt_reports = extract_debt_analysis(state)
              extracted.financials.liquidity = liquidity
              extracted.financials.leverage = leverage
              extracted.financials.debt_structure = debt_struct
              extracted.financials.refinancing_risk = refi
              reports.extend(debt_reports)

              # Phase 6: Audit risk (SECT3-12)
              audit, audit_report = extract_audit_risk(state)
              extracted.financials.audit = audit
              reports.append(audit_report)

              # Phase 7: Tax indicators (SECT3-13)
              tax, tax_report = extract_tax_indicators(state)
              extracted.financials.tax_indicators = tax
              reports.append(tax_report)

              # Phase 8: Peer group (SECT2-09 + SECT3-05)
              peers, peer_report = construct_peer_group(state, override_peers=self._peers)
              extracted.financials.peer_group = peers
              reports.append(peer_report)

              # Phase 9: Financial health narrative (SECT3-01)
              narrative = _generate_financial_narrative(extracted.financials)
              # Store in state or as a field on ExtractedFinancials

              # Validation summary
              _log_validation_summary(reports)

              state.mark_stage_completed(self.name)
          except Exception as exc:
              state.mark_stage_failed(self.name, str(exc))
              raise
  ```

  Helper `_ensure_extracted(state)` creates `ExtractedData` and `ExtractedFinancials` if they don't exist.
  Helper `_get_sector(state)` safely gets sector from company identity.
  Helper `_get_market_cap(state)` safely gets market cap.
  Helper `_log_validation_summary(reports)` logs overall coverage: "Extract stage: N extractors, M total fields expected, K found (X% coverage)".

  **1b. Generate financial health narrative** (SECT3-01):
  `_generate_financial_narrative(financials: ExtractedFinancials) -> SourcedValue[str]`:
  Compose a 3-5 sentence summary covering:

  - Profitability (net income, margins)
  - Liquidity position (current ratio assessment)

  - Key concerns (any distress scores in DISTRESS zone, material weaknesses, going concern)
  Mark as DERIVED/LOW confidence since it's synthesized.

  Add `financial_health_narrative: SourcedValue[str] | None = None` field to `ExtractedFinancials` in `models/financials.py` if not already there.
