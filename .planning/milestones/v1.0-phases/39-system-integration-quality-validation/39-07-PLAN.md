5. **Add Word (.docx) and PDF quality test methods** — per locked decision "All 3 formats must meet the same quality bar":
   ```python
   def test_word_output_quality(self, ticker_output):
       """Verify Word document contains correct company name and populated sections."""
       docx_path = Path(f"output/{ticker_output}/worksheet.docx")
       if not docx_path.exists():
           pytest.skip(f"No Word output for {ticker_output}")
       # Use python-docx to load and extract text
       from docx import Document
       doc = Document(str(docx_path))
       full_text = "\n".join(p.text for p in doc.paragraphs)
       assert "Unknown Company" not in full_text
       # Verify key sections present
       for section in ["Executive Summary", "Company Profile", "Financial", "Governance", "Litigation"]:
           assert section.lower() in full_text.lower(), f"Word missing section: {section}"

   def test_pdf_output_quality(self, ticker_output):
       """Verify PDF contains correct company name and is not empty."""
       pdf_path = Path(f"output/{ticker_output}/worksheet.pdf")
       if not pdf_path.exists():
           pytest.skip(f"No PDF output for {ticker_output}")
       # Verify PDF is non-empty and contains expected content
       assert pdf_path.stat().st_size > 1000, "PDF suspiciously small"
       # If pdfplumber or PyPDF2 available, extract and verify text
       try:
           import pdfplumber
           with pdfplumber.open(str(pdf_path)) as pdf:
               text = "\n".join(page.extract_text() or "" for page in pdf.pages)
           assert "Unknown Company" not in text
       except ImportError:
           pass  # PDF text extraction is best-effort
   ```
   Tests skip gracefully if output files don't exist or extraction libraries aren't available.
