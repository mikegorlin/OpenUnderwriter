   ```python
   """End-to-end pipeline smoke test.

   Validates that the pipeline can run from state.json through RENDER
   and produce all 3 output formats with correct content.

   Skips gracefully if state.json doesn't exist (requires prior pipeline run).
   """
   import pytest
   from pathlib import Path

   TICKERS = ["AAPL"]

   @pytest.fixture(params=TICKERS)
   def ticker(request):
       return request.param

   class TestPipelineSmoke:
       def test_state_file_exists(self, ticker):
           state_path = Path(f"output/{ticker}/state.json")
           if not state_path.exists():
               pytest.skip(f"No state.json for {ticker} — run pipeline first")
           assert state_path.stat().st_size > 1000

       def test_markdown_output_exists_and_correct(self, ticker):
           md_files = list(Path(f"output/{ticker}").glob("*worksheet*.md"))
           if not md_files:
               pytest.skip(f"No Markdown output for {ticker}")
           content = md_files[0].read_text()
           assert len(content) > 1000, "Markdown output suspiciously short"
           assert "Unknown Company" not in content
           assert "Executive Summary" in content or "executive summary" in content.lower()

       def test_word_output_exists(self, ticker):
           docx_files = list(Path(f"output/{ticker}").glob("*.docx"))
           if not docx_files:
               pytest.skip(f"No Word output for {ticker}")
           assert docx_files[0].stat().st_size > 1000, "Word output suspiciously small"

       def test_pdf_output_exists(self, ticker):
           pdf_files = list(Path(f"output/{ticker}").glob("*.pdf"))
           if not pdf_files:
               pytest.skip(f"No PDF output for {ticker}")
           assert pdf_files[0].stat().st_size > 1000, "PDF output suspiciously small"

       def test_all_three_formats_present(self, ticker):
           output_dir = Path(f"output/{ticker}")
           if not output_dir.exists():
               pytest.skip(f"No output directory for {ticker}")
           has_md = bool(list(output_dir.glob("*worksheet*.md")))
           has_docx = bool(list(output_dir.glob("*.docx")))
           has_pdf = bool(list(output_dir.glob("*.pdf")))
           assert has_md, "Missing Markdown output"
           assert has_docx, "Missing Word output"
           assert has_pdf, "Missing PDF output"
   ```

5. **Run the smoke test to verify it passes:**
   ```bash

   ```
