def test_score_override_round_trip(tmp_brain_db):
    """Submit 'this company should be HIGH risk not MEDIUM' and verify scoring adjusts."""
    # 1. Submit feedback: "AAPL should be HIGH risk, current assessment MEDIUM"
    feedback_id = submit_feedback(
        db=tmp_brain_db,
        ticker="AAPL",
        feedback_type="score_override",
        current_value="MEDIUM",
        corrected_value="HIGH",
        rationale="Active SEC investigation not captured in scoring"
    )
    assert feedback_id is not None

    # 2. Verify persistence in DuckDB brain_feedback table
    feedback = get_feedback(db=tmp_brain_db, feedback_id=feedback_id)

    assert feedback.corrected_value == "HIGH"

    # 3. Generate calibration proposal
    proposals = generate_proposals(db=tmp_brain_db)
    assert len(proposals) > 0

    assert len(score_proposal) > 0

    # 4. Apply calibration
