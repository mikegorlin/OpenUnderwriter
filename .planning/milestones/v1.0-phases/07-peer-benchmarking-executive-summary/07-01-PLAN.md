    ```python
    def percentile_rank(company_value: float, peer_values: list[float], higher_is_better: bool = True) -> float:
        """Compute percentile rank (0-100) of company_value within peer_values.

        Uses the standard percentile rank formula:
        rank = ((count_below + 0.5 * count_equal) / N) * 100

        For lower_is_better metrics (debt_to_equity, short_interest), flip:
        the lowest value gets the highest percentile.

        Returns 50.0 if peer_values is empty (no comparison possible).
        """
    ```

    **4. Replace `src/do_uw/stages/benchmark/__init__.py`** (~120-180 lines)

    Replace the stub with the real BenchmarkStage:
    ```python
    class BenchmarkStage:
        @property
        def name(self) -> str:
            return "benchmark"

        def validate_input(self, state: AnalysisState) -> None:
            score = state.stages.get("score")
            if score is None or score.status != StageStatus.COMPLETED:
                msg = "Score stage must be completed before benchmark"
                raise ValueError(msg)

        def run(self, state: AnalysisState) -> None:
            state.mark_stage_running(self.name)

            loader = ConfigLoader()
            brain = loader.load_all()

            # Extract inputs
            scoring = state.scoring  # Must exist (validated)
            company = state.company
            extracted = state.extracted

            # Step 1: Compute peer rankings
            peer_tickers = []
            if extracted and extracted.financials and extracted.financials.peer_group:
                peer_tickers = [p.ticker for p in extracted.financials.peer_group.companies]
            peer_rankings, metric_details = compute_peer_rankings(state, brain.sectors)

            # Step 2: Compute relative position
            relative_position = _compute_relative_position(scoring.quality_score, metric_details)

            # Step 3: Populate BenchmarkResult
            state.benchmark = BenchmarkResult(
                peer_group_tickers=peer_tickers,
                peer_rankings=peer_rankings,
                metric_details=metric_details,
                sector_average_score=_get_sector_avg(brain.sectors, state),
                relative_position=relative_position,
            )

            # Step 4: Compute inherent risk baseline
            sector_code = _get_sector_code(state)
            market_cap = _get_market_cap(state)
            inherent_risk = compute_inherent_risk_baseline(
                sector_code, market_cap,
                scoring.quality_score,
                scoring.tier.tier if scoring.tier else Tier.WRITE,
                brain.sectors, brain.scoring,
            )
            state.benchmark.inherent_risk = inherent_risk

            # Step 5: Build company snapshot (SECT1-01) -- always available
            snapshot = _build_company_snapshot(state)

            # Step 6: Initialize ExecutiveSummary with snapshot + inherent risk + deal context
            # (key_findings and thesis populated in summary_builder.py, Plan 07-02)
            state.executive_summary = ExecutiveSummary(
                snapshot=snapshot,
                inherent_risk=inherent_risk,
                deal_context=DealContext(),
            )

            state.mark_stage_completed(self.name)
    ```

    Helper functions for extracting sector code, market cap, and building snapshot from state.
