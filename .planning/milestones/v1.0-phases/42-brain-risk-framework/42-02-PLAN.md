```python
def render_peril_summary(
    doc: Any, peril_data: dict[str, Any], ds: DesignSystem
) -> None:
    """Render D&O Peril Summary table — overview of all 8 perils."""
    heading = doc.add_paragraph(style="DOHeading3")
    heading.add_run("D&O Claim Peril Assessment")

    all_perils = peril_data.get("all_perils", [])
    if not all_perils:
        para = doc.add_paragraph(style="DOBody")
        para.add_run("Peril assessment not available.")
        return

    # Summary stats
    active_count = peril_data.get("active_count", 0)
    intro = doc.add_paragraph(style="DOBody")
    intro.add_run(
        f"{active_count} of 8 D&O claim perils show active risk signals. "
        f"Active perils have at least one causal chain with triggered checks."
    )

    # Table: Peril | Risk Level | Active Chains | Key Evidence
    headers = ["Peril", "Risk Level", "Active Chains", "Key Evidence"]
    rows = []
    for peril in all_perils:
        risk = peril["risk_level"]
        chains = f"{peril['active_chain_count']}/{peril['total_chain_count']}"
        evidence = "; ".join(peril.get("key_evidence", [])[:2]) or "—"
        if len(evidence) > 60:
            evidence = evidence[:57] + "..."
        rows.append([peril["name"], risk, chains, evidence])

    add_styled_table(doc, headers, rows, ds)
```

**Per-peril deep dives** — causal chain narratives:

```python
def render_peril_deep_dives(
    doc: Any, peril_data: dict[str, Any], ds: DesignSystem
) -> None:
    """Render deep dive for each active peril with causal chain narratives."""
    active_perils = peril_data.get("perils", [])
    if not active_perils:
        return

    heading = doc.add_paragraph(style="DOHeading3")
    heading.add_run("Active Peril Analysis")

    for peril in active_perils:
        _render_single_peril(doc, peril, ds)


def _render_single_peril(doc: Any, peril: dict, ds: DesignSystem) -> None:
    """Render a single peril with its active causal chains."""
    # Peril header with risk indicator
    sub_heading = doc.add_paragraph(style="DOHeading3")
    sub_heading.add_run(f"{peril['name']}: {peril['risk_level']}")
    add_risk_indicator(sub_heading, peril["risk_level"], ds)

    # Context line
    ctx = doc.add_paragraph(style="DOBody")
    freq = peril.get("frequency", "unknown")
    sev = peril.get("severity", "unknown")
    settlement = peril.get("typical_settlement_range", "")
    ctx.add_run(
        f"Frequency: {freq} | Severity: {sev}"
        + (f" | Typical range: {settlement}" if settlement else "")
    )

    # Each active chain
    for chain in peril.get("chains", []):
        if not chain.get("active"):
            continue
        _render_chain_narrative(doc, chain, ds)


def _render_chain_narrative(doc: Any, chain: dict, ds: DesignSystem) -> None:
    """Render a causal chain as trigger → amplifier → mitigator narrative."""
    # Chain name + risk level
    chain_para = doc.add_paragraph(style="DOBody")
    name_run = chain_para.add_run(f"Chain: {chain['name']}")
    name_run.bold = True

    # Description
    if chain.get("description"):
        desc_para = doc.add_paragraph(style="DOBody")
        desc_run = desc_para.add_run(chain["description"])
        desc_run.italic = True
        desc_run.font.size = ds.size_small

    # Triggers (what started it)
    triggers = chain.get("triggered_triggers", [])
    if triggers:
        t_para = doc.add_paragraph(style="DOBody")
        t_para.add_run("  Triggers: ").bold = True
        for t in triggers:
            evidence = chain.get("evidence_map", {}).get(t, "")
            line = f"{t}"
            if evidence:
                line += f" — {evidence}"
            t_para.add_run(f"\n    - {line}")

    # Amplifiers (what makes it worse)
    amplifiers = chain.get("active_amplifiers", [])
    if amplifiers:
        a_para = doc.add_paragraph(style="DOBody")
        a_para.add_run("  Amplifiers: ").bold = True
        for a in amplifiers:
            a_para.add_run(f"\n    - {a}")

    # Mitigators (what helps)
    mitigators = chain.get("active_mitigators", [])
    if mitigators:
        m_para = doc.add_paragraph(style="DOBody")
        m_para.add_run("  Mitigators: ").bold = True
        for m in mitigators:
            m_para.add_run(f"\n    - {m}")

    # Historical context
    filing_rate = chain.get("historical_filing_rate")
    median_sev = chain.get("median_severity_usd")
    if filing_rate or median_sev:
        hist = doc.add_paragraph(style="DOBody")
        parts = []
        if filing_rate:
            parts.append(f"Historical filing rate: {filing_rate*100:.0f}%")
        if median_sev:
            parts.append(f"Median severity: ${median_sev/1e6:.0f}M")
        hist_run = hist.add_run("  " + " | ".join(parts))
        hist_run.font.size = ds.size_small
        hist_run.font.color.rgb = ds.color_text_light
```
