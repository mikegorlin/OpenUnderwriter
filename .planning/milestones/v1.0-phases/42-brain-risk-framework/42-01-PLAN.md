```python
def load_perils(self) -> list[dict[str, Any]]:
    """Load all 8 perils from brain_perils table."""
    conn = self._get_conn()
    rows = conn.execute(
        "SELECT peril_id, name, description, haz_codes, frequency, severity, "
        "typical_settlement_range, key_drivers FROM brain_perils"
    ).fetchall()
    cols = ["peril_id", "name", "description", "haz_codes", "frequency",
            "severity", "typical_settlement_range", "key_drivers"]
    return [dict(zip(cols, row)) for row in rows]

def load_causal_chains(self) -> list[dict[str, Any]]:
    """Load all causal chains from brain_causal_chains table."""
    conn = self._get_conn()
    rows = conn.execute(
        "SELECT chain_id, name, peril_id, description, "
        "trigger_checks, amplifier_checks, mitigator_checks, evidence_checks, "
        "frequency_factors, severity_factors, patterns, red_flags, "
        "historical_filing_rate, median_severity_usd "
        "FROM brain_causal_chains"
    ).fetchall()
    cols = ["chain_id", "name", "peril_id", "description",
            "trigger_checks", "amplifier_checks", "mitigator_checks", "evidence_checks",
            "frequency_factors", "severity_factors", "patterns", "red_flags",
            "historical_filing_rate", "median_severity_usd"]
    return [dict(zip(cols, row)) for row in rows]
```

**Chain activation logic:**
```python
def _evaluate_chain(chain: dict, check_results: dict) -> dict:
    """Evaluate a causal chain against pipeline check results.

    Returns chain status dict with:
    - active: bool (any trigger fired)
    - risk_level: HIGH/ELEVATED/MODERATE/LOW
    - triggered_triggers: list of check_ids that fired
    - active_amplifiers: list of amplifier check_ids that fired
    - active_mitigators: list of mitigator check_ids that fired
    - evidence_summary: list of evidence strings
    """
    triggers = chain.get("trigger_checks", [])
    amplifiers = chain.get("amplifier_checks", [])
    mitigators = chain.get("mitigator_checks", [])

    fired_triggers = [c for c in triggers if _check_fired(c, check_results)]
    fired_amplifiers = [c for c in amplifiers if _check_fired(c, check_results)]
    fired_mitigators = [c for c in mitigators if _check_fired(c, check_results)]

    # Chain is active if any trigger check fired RED or YELLOW
    active = len(fired_triggers) > 0

    # Risk level
    if len(fired_triggers) >= 2 or _has_associated_red_flag(chain, check_results):
        risk_level = "HIGH"
    elif fired_triggers and fired_amplifiers:
        risk_level = "ELEVATED"
    elif fired_triggers:
        risk_level = "MODERATE"
    else:
        risk_level = "LOW"

    # Mitigators can reduce one level
    if fired_mitigators and risk_level in ("HIGH", "ELEVATED"):
        risk_level = {"HIGH": "ELEVATED", "ELEVATED": "MODERATE"}[risk_level]

    return {...}
```

**Check fired detection:**
```python
def _check_fired(check_id: str, check_results: dict) -> bool:
    """Check if a specific check fired (RED or YELLOW status)."""
    result = check_results.get(check_id)
    if not isinstance(result, dict):
        return False
    status = result.get("status", "").upper()
    return status in ("RED", "YELLOW", "TRIGGERED", "ELEVATED")
```

**Per-peril aggregation:**
```python
def _aggregate_peril(peril: dict, chains: list[dict], check_results: dict) -> dict:
    """Aggregate chain results into a peril-level assessment."""
    chain_results = [_evaluate_chain(c, check_results) for c in chains]
    active_chains = [c for c in chain_results if c["active"]]

    # Peril risk = highest chain risk
    risk_levels = [c["risk_level"] for c in active_chains]
    peril_risk = max(risk_levels, key=_risk_order, default="LOW")

    return {
        "peril_id": peril["peril_id"],
        "name": peril["name"],
        "description": peril["description"],
        "risk_level": peril_risk,
        "active_chain_count": len(active_chains),
        "total_chain_count": len(chains),
        "chains": chain_results,
        "frequency": peril.get("frequency", "unknown"),
        "severity": peril.get("severity", "unknown"),
        "typical_settlement_range": peril.get("typical_settlement_range", ""),
        "key_evidence": _collect_peril_evidence(active_chains),
    }
```

**Top-level extraction:**
```python
def extract_peril_scoring(state: AnalysisState) -> dict[str, Any]:
    """Extract peril-organized scoring data for templates.

    Returns dict with:
    - perils: list of peril assessments (only active ones)
    - all_perils: list of all 8 perils with risk levels
    - active_count: number of perils with active chains
    - highest_peril: peril_id with highest risk
    """
    try:
        from do_uw.brain.brain_loader import BrainDBLoader
        loader = BrainDBLoader()
        perils = loader.load_perils()
        chains = loader.load_causal_chains()
    except Exception:
        return {}  # Graceful fallback

    check_results = {}
    if state.analysis and state.analysis.check_results:
        check_results = state.analysis.check_results

    # Group chains by peril
    chains_by_peril = {}
    for chain in chains:
        pid = chain["peril_id"]
        chains_by_peril.setdefault(pid, []).append(chain)

    # Evaluate each peril
    peril_results = []
    for peril in perils:
        pid = peril["peril_id"]
        peril_chains = chains_by_peril.get(pid, [])
        result = _aggregate_peril(peril, peril_chains, check_results)
        peril_results.append(result)

    # Sort: active perils first, then by risk level
    active = [p for p in peril_results if p["active_chain_count"] > 0]
    active.sort(key=lambda p: _risk_order(p["risk_level"]), reverse=True)

    highest = active[0]["peril_id"] if active else None

    return {
        "perils": active,
        "all_perils": peril_results,
        "active_count": len(active),
        "highest_peril": highest,
    }
```
